93bool MCPAuthorizationManager::check_access(
94 const std::string& identity_name,
95 const std::string& tool_name,
96 MCPAccessLevel required_level) const {
97 std::lock_guard<std::mutex> lock(auth_mutex_);
98 auto it = key_sets_.find(identity_name);
99 if (it == key_sets_.end()) {
100 logger->info("Auth ALLOW: {} on {} (no enforcement)",
101 identity_name, tool_name);
102 return true;
103 }
104 bool allowed = it->second.has_access(tool_name, required_level);
105 logger->info("Auth {}: {} on {} (required={})",
106 allowed ? "ALLOW" : "DENY",
107 identity_name, tool_name,
108 static_cast<int>(required_level));
109 return allowed;
110}

122entropic_error_t MCPAuthorizationManager::grant_from(
123 const std::string& granter_name,
124 const std::string& grantee_name,
125 const std::string& pattern,
126 MCPAccessLevel level) {
127 std::lock_guard<std::mutex> lock(auth_mutex_);
128 auto granter = key_sets_.find(granter_name);
129 auto grantee = key_sets_.find(grantee_name);
130 if (granter == key_sets_.end() ||
131 grantee == key_sets_.end()) {
132 return ENTROPIC_ERROR_IDENTITY_NOT_FOUND;
133 }
134 if (!granter->second.has_access(pattern, level)) {
135 logger->warn("{} lacks {}:{} — cannot grant to {}",
136 granter_name, pattern,
137 mcp_access_level_name(level), grantee_name);
138 return ENTROPIC_ERROR_PERMISSION_DENIED;
139 }
140 grantee->second.grant(pattern, level);
141 logger->info("{} granted {}:{} to {}", granter_name, pattern,
142 mcp_access_level_name(level), grantee_name);
143 return ENTROPIC_OK;
144}

186bool MCPAuthorizationManager::deserialize_all(
187 const std::string& json) {
188 nlohmann::json obj;
189 try {
190 obj = nlohmann::json::parse(json);
191 } catch (const nlohmann::json::exception& e) {
192 logger->warn("Deserialize failed: {}", e.what());
193 return false;
194 }
195 if (!obj.is_object()) {
196 logger->warn("Deserialize: expected JSON object");
197 return false;
198 }
199 std::lock_guard<std::mutex> lock(auth_mutex_);
200 key_sets_.clear();
201 for (const auto& [name, keys_arr] : obj.items()) {
202 auto [it, _] = key_sets_.try_emplace(name);
203 it->second.deserialize(keys_arr.dump());
204 }
205 return true;
206}
