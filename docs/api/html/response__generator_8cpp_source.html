94void ResponseGenerator::lock_tier_if_needed(LoopContext& ctx) {
95 if (!ctx.locked_tier.empty()) {
96 if (callbacks_.on_tier_selected != nullptr) {
97 callbacks_.on_tier_selected(ctx.locked_tier.c_str(),
98 callbacks_.user_data);
99 }
100 return;
101 }
102
103 if (inference_.route == nullptr) {
104 ctx.locked_tier = "default";
105 return;
106 }
107
108 auto msgs_json = serialize_messages(ctx.messages);
109 char* result_json = nullptr;
110 int rc = inference_.route(msgs_json.c_str(), &result_json,
111 inference_.orchestrator_data);
112 if (rc == 0 && result_json != nullptr) {
113 ctx.locked_tier = result_json;
114 if (inference_.free_fn != nullptr) {
115 inference_.free_fn(result_json);
116 }
117 } else {
118 ctx.locked_tier = "default";
119 logger->warn("Routing failed (rc={}), using default tier", rc);
120 }
121
122 logger->info("Locked tier: {}", ctx.locked_tier);
123 if (callbacks_.on_tier_selected != nullptr) {
124 callbacks_.on_tier_selected(ctx.locked_tier.c_str(),
125 callbacks_.user_data);
126 }
127}

153static void stream_token_callback(
154 const char* token,
155 size_t len,
156 void* user_data) {
157 auto* acc = static_cast<StreamAccumulator*>(user_data);
158 if (acc->events->interrupt != nullptr
159 && acc->events->interrupt->load()) {
160 acc->interrupted = true;
161 return;
162 }
163 acc->content.append(token, len);
164 if (acc->callbacks->on_stream_chunk != nullptr) {
165 acc->callbacks->on_stream_chunk(token, len,
166 acc->callbacks->user_data);
167 }
168
169 // Hook: ON_STREAM_TOKEN (v1.9.1)
170 if (acc->hooks != nullptr && acc->hooks->fire_info != nullptr) {
171 std::string json = "{\"token_index\":"
172 + std::to_string(acc->token_index++) + "}";
173 acc->hooks->fire_info(acc->hooks->registry,
174 ENTROPIC_HOOK_ON_STREAM_TOKEN, json.c_str());
175 }
176}

185GenerateResult ResponseGenerator::generate_streaming(LoopContext& ctx) {
186 if (inference_.generate_stream == nullptr) {
187 logger->warn("No streaming function, falling back to batch");
188 return generate_batch(ctx);
189 }
190
191 logger->info("Generate (stream): tier={}, {} messages",
192 ctx.locked_tier, ctx.messages.size());
193 log_prompt(ctx.messages, ctx.locked_tier);
194 auto msgs_json = serialize_messages(ctx.messages);
195 auto params_json = build_params_json();
196
197 int cancel_flag = 0;
198 StreamAccumulator acc;
199 acc.callbacks = &callbacks_;
200 acc.events = &events_;
201 acc.hooks = &hooks_;
202
203 inference_.generate_stream(
204 msgs_json.c_str(), params_json.c_str(),
205 stream_token_callback, &acc,
206 &cancel_flag, inference_.backend_data);
207
208 GenerateResult result;
209 result.finish_reason = acc.interrupted ? "interrupted" : "stop";
210 result.content = acc.content;
211 result.tool_calls_json = "[]";
212 logger->info("Generate complete (stream): finish={}, {} chars",
213 result.finish_reason, result.content.size());
214 return result;
215}

224GenerateResult ResponseGenerator::generate_batch(LoopContext& ctx) {
225 if (inference_.generate == nullptr) {
226 logger->error("No generate function available");
227 return {"", "[]", "error"};
228 }
229
230 logger->info("Generate (batch): tier={}, {} messages",
231 ctx.locked_tier, ctx.messages.size());
232 log_prompt(ctx.messages, ctx.locked_tier);
233 auto msgs_json = serialize_messages(ctx.messages);
234 auto params_json = build_params_json();
235 char* result_json = nullptr;
236
237 int rc = inference_.generate(
238 msgs_json.c_str(), params_json.c_str(),
239 &result_json, inference_.backend_data);
240
241 GenerateResult result;
242 if (rc == 0 && result_json != nullptr) {
243 result.content = result_json;
244 result.finish_reason = "stop";
245 result.tool_calls_json = "[]";
246 if (inference_.free_fn != nullptr) {
247 inference_.free_fn(result_json);
248 }
249 } else {
250 result.finish_reason = "error";
251 logger->error("Generate failed (rc={})", rc);
252 }
253 logger->info("Generate complete (batch): finish={}, {} chars",
254 result.finish_reason, result.content.size());
255 return result;
256}

266std::string ResponseGenerator::handle_pause(
267 LoopContext& ctx,
268 const std::string& partial) {
269 ctx.state = AgentState::PAUSED;
270 if (callbacks_.on_state_change != nullptr) {
271 callbacks_.on_state_change(
272 static_cast<int>(AgentState::PAUSED),
273 callbacks_.user_data);
274 }
275
276 char* injection = nullptr;
277 if (callbacks_.on_pause_prompt != nullptr) {
278 callbacks_.on_pause_prompt(partial.c_str(), &injection,
279 callbacks_.user_data);
280 }
281
282 if (injection == nullptr) {
283 if (events_.interrupt != nullptr) {
284 events_.interrupt->store(true);
285 }
286 return partial;
287 }
288
289 std::string inj(injection);
290 if (inj.empty()) {
291 ctx.state = AgentState::EXECUTING;
292 return partial;
293 }
294
295 // Injection provided: append partial + injection to messages
296 if (!partial.empty()) {
297 Message partial_msg;
298 partial_msg.role = "assistant";
299 partial_msg.content = partial + "\n\n[Generation paused by user]";
300 ctx.messages.push_back(std::move(partial_msg));
301 }
302 Message inject_msg;
303 inject_msg.role = "user";
304 inject_msg.content = "[User interjection]: " + inj
305 + "\n\nPlease continue with this in mind.";
306 ctx.messages.push_back(std::move(inject_msg));
307
308 ctx.state = AgentState::EXECUTING;
309 return "";
310}

319std::string ResponseGenerator::serialize_messages(
320 const std::vector<Message>& messages) {
321 // Minimal JSON serialization — no nlohmann/json dependency.
322 std::string json = "[";
323 for (size_t i = 0; i < messages.size(); ++i) {
324 if (i > 0) { json += ","; }
325 json += "{\"role\":\"" + messages[i].role
326 + "\",\"content\":\"";
327 // Escape content for JSON
328 for (char c : messages[i].content) {
329 switch (c) {
330 case '"': json += "\\\""; break;
331 case '\\': json += "\\\\"; break;
332 case '\n': json += "\\n"; break;
333 case '\r': json += "\\r"; break;
334 case '\t': json += "\\t"; break;
335 default: json += c; break;
336 }
337 }
338 json += "\"}";
339 }
340 json += "]";
341 return json;
342}
