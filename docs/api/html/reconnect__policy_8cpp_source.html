53uint32_t ReconnectPolicy::delay_ms(uint32_t attempt) const {
54 double raw = static_cast<double>(base_delay_ms_) *
55 std::pow(backoff_factor_, static_cast<double>(attempt));
56 double capped = std::min(raw, static_cast<double>(max_delay_ms_));
57
58 // Add jitter: uniform random in [0, 10% of delay]
59 thread_local std::mt19937 rng(std::random_device{}());
60 double jitter_range = capped * 0.1;
61 std::uniform_real_distribution<double> dist(0.0, jitter_range);
62
63 return static_cast<uint32_t>(capped + dist(rng));
64}
