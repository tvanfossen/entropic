47size_t GrammarRegistry::load_bundled(
48 const std::filesystem::path& grammar_dir)
49{
50 if (!std::filesystem::is_directory(grammar_dir)) {
51 logger->warn("Grammar directory not found: {}",
52 grammar_dir.string());
53 return 0;
54 }
55
56 size_t count = 0;
57 for (const auto& entry : std::filesystem::directory_iterator(grammar_dir)) {
58 if (entry.path().extension() != ".gbnf") {
59 continue;
60 }
61
62 std::string key = entry.path().stem().string();
63 std::string content = read_file_contents(entry.path());
64 if (content.empty()) {
65 logger->warn("Empty or unreadable grammar file: {}",
66 entry.path().string());
67 continue;
68 }
69
70 std::string error = validate(content);
71 GrammarEntry ge;
72 ge.key = key;
73 ge.gbnf_content = std::move(content);
74 ge.source = "bundled";
75 ge.validated = error.empty();
76 ge.error = std::move(error);
77
78 if (!ge.validated) {
79 logger->warn("Bundled grammar '{}' has validation errors: {}",
80 key, ge.error);
81 }
82
83 std::lock_guard<std::mutex> lock(registry_mutex_);
84 grammars_[key] = std::move(ge);
85 ++count;
86 }
87
88 logger->info("Loaded {} bundled grammar(s)", count);
89 return count;
90}

101bool GrammarRegistry::register_grammar(
102 const std::string& key,
103 const std::string& gbnf_content,
104 const std::string& source)
105{
106 std::lock_guard<std::mutex> lock(registry_mutex_);
107 if (grammars_.count(key) > 0) {
108 logger->warn("Grammar key '{}' already registered", key);
109 return false;
110 }
111
112 std::string error = validate(gbnf_content);
113 GrammarEntry ge;
114 ge.key = key;
115 ge.gbnf_content = gbnf_content;
116 ge.source = source;
117 ge.validated = error.empty();
118 ge.error = std::move(error);
119
120 grammars_[key] = std::move(ge);
121 logger->info("Registered grammar '{}' (source={}, valid={})",
122 key, source, grammars_[key].validated);
123 return true;
124}

239std::string GrammarRegistry::validate(const std::string& gbnf_content) {
240 if (gbnf_content.empty()) {
241 return "empty grammar string";
242 }
243
244 // Use llama_sampler_init_grammar with nullptr vocab for validation.
245 // Grammar parsing is independent of vocabulary — the GBNF parser
246 // only needs the grammar string. Returns nullptr on parse failure.
247 llama_sampler* sampler = llama_sampler_init_grammar(
248 nullptr, gbnf_content.c_str(), "root");
249
250 if (sampler == nullptr) {
251 return "GBNF parse failed";
252 }
253
254 llama_sampler_free(sampler);
255 return "";
256}
