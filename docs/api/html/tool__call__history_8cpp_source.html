54std::vector<ToolCallRecord> ToolCallHistory::recent(size_t count) const {
55 std::shared_lock lock(mutex_);
56 size_t n = std::min(count, count_);
57 std::vector<ToolCallRecord> result;
58 result.reserve(n);
59
60 for (size_t i = 0; i < n; ++i) {
61 size_t idx = (head_ + capacity_ - 1 - i) % capacity_;
62 result.push_back(buffer_[idx]);
63 }
