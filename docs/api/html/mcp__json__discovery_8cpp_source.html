71void MCPJsonDiscovery::parse_mcp_json(
72 const std::filesystem::path& path,
73 const std::filesystem::path& own_socket,
74 const std::set<std::string>& existing_names,
75 std::set<std::string>& seen,
76 std::vector<ExternalServerConfig>& out) const {
77
78 if (!std::filesystem::exists(path)) {
79 return;
80 }
81
82 nlohmann::json data;
83 try {
84 std::ifstream f(path);
85 data = nlohmann::json::parse(f);
86 } catch (const std::exception& e) {
87 logger->warn("Failed to read {}: {}", path.string(), e.what());
88 return;
89 }
90
91 auto servers_it = data.find("mcpServers");
92 if (servers_it == data.end() || !servers_it->is_object()) {
93 return;
94 }
95
96 for (auto& [name, cfg] : servers_it->items()) {
97 parse_server_entry(name, cfg, own_socket,
98 existing_names, seen, out);
99 }
100}

113void MCPJsonDiscovery::parse_server_entry(
114 const std::string& name,
115 const nlohmann::json& cfg,
116 const std::filesystem::path& own_socket,
117 const std::set<std::string>& existing_names,
118 std::set<std::string>& seen,
119 std::vector<ExternalServerConfig>& out) const {
120
121 // Skip if already discovered from a higher-priority file
122 if (seen.count(name) > 0) {
123 return;
124 }
125
126 // Self-detection: skip own socket
127 if (cfg.contains("path")) {
128 auto sock = std::filesystem::path(
129 cfg["path"].get<std::string>());
130 if (std::filesystem::weakly_canonical(sock) ==
131 std::filesystem::weakly_canonical(own_socket)) {
132 logger->debug("Skipping '{}' (matches own socket)", name);
133 seen.insert(name);
134 return;
135 }
136 }
137
138 // Shadow warning: skip if already registered
139 if (existing_names.count(name) > 0) {
140 logger->warn(".mcp.json entry '{}' shadowed by existing "
141 "config — skipping", name);
142 seen.insert(name);
143 return;
144 }
145
146 // Determine transport
147 ExternalServerConfig entry;
148 entry.name = name;
149 std::string type = cfg.value("type", "");
150
151 if (infer_transport(cfg, type, entry)) {
152 seen.insert(name);
153 out.push_back(std::move(entry));
154 logger->info("Discovered external server '{}' "
155 "(transport={})", name, entry.transport);
156 }
157}

168bool MCPJsonDiscovery::infer_transport(
169 const nlohmann::json& cfg,
170 const std::string& type,
171 ExternalServerConfig& entry) const {
172
173 bool has_url = cfg.contains("url");
174 bool has_command = cfg.contains("command");
175
176 // Explicit type overrides inference
177 if (type == "sse" || type == "http" ||
178 (type.empty() && has_url)) {
179 return parse_sse_entry(cfg, entry);
180 }
181 if (type == "stdio" || (type.empty() && has_command)) {
182 return parse_stdio_entry(cfg, entry);
183 }
184
185 logger->warn(".mcp.json entry '{}': cannot determine transport "
186 "(no url or command)", entry.name);
187 return false;
188}

220bool MCPJsonDiscovery::parse_stdio_entry(
221 const nlohmann::json& cfg,
222 ExternalServerConfig& entry) const {
223
224 entry.command = cfg.value("command", "");
225 if (entry.command.empty()) {
226 logger->warn(".mcp.json entry '{}' missing 'command' — "
227 "skipping", entry.name);
228 return false;
229 }
230 entry.transport = "stdio";
231
232 // Args
233 if (cfg.contains("args") && cfg["args"].is_array()) {
234 for (const auto& arg : cfg["args"]) {
235 entry.args.push_back(arg.get<std::string>());
236 }
237 }
238
239 // Env with blocklist enforcement
240 if (cfg.contains("env") && cfg["env"].is_object()) {
241 for (auto& [key, val] : cfg["env"].items()) {
242 if (is_blocked_env_var(key)) {
243 logger->warn(".mcp.json '{}': blocked env var '{}' "
244 "— skipping", entry.name, key);
245 continue;
246 }
247 entry.env[key] = val.get<std::string>();
248 }
249 }
250 return true;
251}

260std::filesystem::path compute_socket_path(
261 const std::filesystem::path& project_dir) {
262
263 auto resolved = std::filesystem::weakly_canonical(project_dir);
264 auto input = resolved.string();
265
266 // SHA-256 first 8 hex chars (simplified: use std::hash)
267 auto h = std::hash<std::string>{}(input);
268 std::ostringstream ss;
269 ss << std::hex << std::setfill('0') << std::setw(16) << h;
270 auto hash_str = ss.str().substr(0, 8);
271
272 auto home = std::filesystem::path(
273 getenv("HOME") ? getenv("HOME") : "/tmp");
274 return home / ".entropic" / "socks" / (hash_str + ".sock");
275}

284static bool has_blocked_prefix(const std::string& key) {
285 return (key.size() >= 9 && key.substr(0, 9) == "ENTROPIC_") ||
286 (key.size() >= 5 && key.substr(0, 5) == "DYLD_");

296bool is_blocked_env_var(const std::string& key) {
297 static const std::array<std::string, 7> blocked = {{
298 "LD_PRELOAD", "LD_LIBRARY_PATH", "PATH",
299 "HOME", "SHELL", "DYLD_INSERT_LIBRARIES",
300 "DYLD_LIBRARY_PATH"
301 }};
302
303 bool exact_match = std::any_of(blocked.begin(), blocked.end(),
304 [&key](const std::string& b) { return key == b; });
305
306 return exact_match || has_blocked_prefix(key);
307}
