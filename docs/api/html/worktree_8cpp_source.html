31GitResult run_git(const std::filesystem::path& repo_dir,
32 const std::string& args) {
33 std::string cmd = "cd '" + repo_dir.string() +
34 "' && git " + args + " 2>&1";
35
36 GitResult result;
37 std::array<char, 256> buf{};
38
39 FILE* pipe = popen(cmd.c_str(), "r");
40 if (pipe == nullptr) {
41 logger->error("popen failed for: git {}", args);
42 return result;
43 }
44
45 while (fgets(buf.data(), static_cast<int>(buf.size()), pipe) != nullptr) {
46 result.output += buf.data();
47 }
48
49 int status = pclose(pipe);
50 result.success = (status == 0);
51 return result;
52}

132std::optional<WorktreeInfo> WorktreeManager::create_worktree(
133 const std::string& delegation_id,
134 const std::string& tier) {
135 if (!develop_ready_) {
136 logger->error("Cannot create worktree: develop not ready");
137 return std::nullopt;
138 }
139
140 std::string short_id = delegation_id.substr(
141 0, std::min(delegation_id.size(), size_t{8}));
142 std::string branch = "delegation/" + tier + "-" + short_id;
143 auto path = worktree_base_ / ("delegation-" + short_id);
144
145 std::string cmd = "worktree add -b " + branch +
146 " '" + path.string() + "' develop";
147 auto r = run_git(repo_dir_, cmd);
148 if (!r.success) {
149 logger->error("Failed to create worktree: {}", r.output);
150 return std::nullopt;
151 }
152
153 logger->info("Created worktree: {} -> {}", branch, path.string());
154 return WorktreeInfo{path, branch, short_id};

232bool WorktreeManager::accept_to_main() {
233 std::string main_branch = find_main_branch();
234 if (main_branch.empty()) {
235 logger->error("No main/master branch found");
236 return false;
237 }
238
239 auto co = run_git(repo_dir_, "checkout " + main_branch);
240 if (!co.success) {
241 return false;
242 }
243
244 auto merge = run_git(repo_dir_, "merge --no-ff develop "
245 "-m 'Accept develop into " +
246 main_branch + "'");
247 run_git(repo_dir_, "checkout develop");
248
249 if (!merge.success) {
250 logger->error("accept_to_main failed: {}", merge.output);
251 }
252 return merge.success;
253}
