80ToolDefinition load_tool_definition(
81 const std::string& tool_name,
82 const std::string& server_prefix,
83 const std::string& data_dir) {
84
85 std::string path = data_dir;
86 if (!server_prefix.empty()) {
87 path += "/" + server_prefix;
88 }
89 path += "/" + tool_name + ".json";
90
91 std::ifstream file(path);
92 if (!file.is_open()) {
93 throw std::runtime_error(
94 "Tool definition not found: " + path);
95 }
96
97 auto json = nlohmann::json::parse(file);
98
99 ToolDefinition def;
100 def.name = json.at("name").get<std::string>();
101 def.description = json.value("description", "");
102 def.input_schema = json.at("inputSchema").dump();
103
104 logger->info("Loaded tool definition: {} from {}",
105 def.name, path);
106 return def;
107}
