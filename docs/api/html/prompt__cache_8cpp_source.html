90bool PromptCache::store(const CacheKey& key,
91 std::vector<uint8_t>&& data,
92 int token_count)
93{
94 std::lock_guard<std::mutex> lock(mutex_);
95
96 size_t entry_size = data.size();
97
98 if (max_bytes_ == 0) {
99 return false;
100 }
101
102 if (entry_size > max_bytes_) {
103 logger->warn("Cache entry ({} bytes) exceeds max_bytes ({}), skipping",
104 entry_size, max_bytes_);
105 return false;
106 }
107
108 // Remove existing entry with same key if present
109 auto it = entries_.find(key);
110 if (it != entries_.end()) {
111 bytes_used_ -= it->second.data_size;
112 entries_.erase(it);
113 auto lru_it = lru_map_.find(key);
114 if (lru_it != lru_map_.end()) {
115 lru_.erase(lru_it->second);
116 lru_map_.erase(lru_it);
117 }
118 }
119
120 evict_until(entry_size);
121
122 CacheEntry entry;
123 entry.data = std::move(data);
124 entry.token_count = token_count;
125 entry.data_size = entry_size;
126
127 entries_[key] = std::move(entry);
128 lru_.push_front(key);
129 lru_map_[key] = lru_.begin();
130
131 bytes_used_ += entry_size;
132 ++stats_.stores;
133 if (bytes_used_ > stats_.peak_bytes) {
134 stats_.peak_bytes = bytes_used_;
135 }
136
137 logger->info("Cached prompt prefix: {} bytes, {} tokens, {} entries total",
138 entry_size, token_count, entries_.size());
139 return true;

153const CacheEntry* PromptCache::lookup(const CacheKey& key) {
154 std::lock_guard<std::mutex> lock(mutex_);
155
156 auto it = entries_.find(key);
157 if (it == entries_.end()) {
158 ++stats_.misses;
159 logger->info("Cache MISS: key={:016x}, {} hits / {} misses",
160 key.hash, stats_.hits, stats_.misses);
161 return nullptr;
162 }
163
164 // Move to front of LRU
165 auto lru_it = lru_map_.find(key);
166 if (lru_it != lru_map_.end()) {
167 lru_.splice(lru_.begin(), lru_, lru_it->second);
168 }
169
170 ++stats_.hits;
171 logger->info("Cache HIT: key={:016x}, {} tokens, {} bytes",
172 key.hash, it->second.token_count,
173 it->second.data_size);
174 return &it->second;
175}

234void PromptCache::evict_until(size_t needed_bytes) {
235 while (bytes_used_ + needed_bytes > max_bytes_ && !lru_.empty()) {
236 CacheKey victim = lru_.back();
237 lru_.pop_back();
238 lru_map_.erase(victim);
239
240 auto it = entries_.find(victim);
241 if (it != entries_.end()) {
242 bytes_used_ -= it->second.data_size;
243 ++stats_.evictions;
244 logger->info("Evicted cache entry: {} bytes freed",
245 it->second.data_size);
246 entries_.erase(it);
247 }
