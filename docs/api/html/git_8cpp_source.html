32static std::pair<std::string, int> run_git(
33 const std::string& repo_dir,
34 const std::string& git_args) {
35
36 std::string full_cmd =
37 "git -C " + repo_dir + " " + git_args + " 2>&1";
38
39 FILE* pipe = popen(full_cmd.c_str(), "r"); // NOLINT
40 if (pipe == nullptr) {
41 return {"Failed to open git process", -1};
42 }
43
44 std::string output;
45 std::array<char, 4096> buf{};
46 while (fgets(buf.data(), buf.size(), pipe) != nullptr) {
47 output += buf.data();
48 }
49
50 int status = pclose(pipe); // NOLINT
51 int exit_code = WEXITSTATUS(status);
52 logger->info("Git: '{}' exit={}, {} chars",
53 git_args, exit_code, output.size());
54 return {output, exit_code};
55}

455 load_tool_definition("diff", "git", tools_dir), *this);
456 log_ = std::make_unique<GitLogTool>(
457 load_tool_definition("log", "git", tools_dir), *this);

459 load_tool_definition("commit", "git", tools_dir), *this);

462 checkout_ = std::make_unique<GitCheckoutTool>(
463 load_tool_definition("checkout", "git", tools_dir), *this);
464 add_ = std::make_unique<GitAddTool>(
465 load_tool_definition("add", "git", tools_dir), *this);
466 reset_ = std::make_unique<GitResetTool>(
467 load_tool_definition("reset", "git", tools_dir), *this);
468
469 register_tool(status_.get());

471 register_tool(log_.get());
472 register_tool(commit_.get());
473 register_tool(branch_.get());
474 register_tool(checkout_.get());
475 register_tool(add_.get());
476 register_tool(reset_.get());

479 repo_dir_.string());
