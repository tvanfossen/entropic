209bool glob_match(const std::string& filename,
210 const std::string& pattern) {
211 std::string regex_str;
212 regex_str.reserve(pattern.size() * 2);
213
214 for (char ch : pattern) {
215 if (ch == '*') {
216 regex_str += ".*";
217 } else if (ch == '?') {
218 regex_str += '.';
219 } else if (ch == '.') {
220 regex_str += "\\.";
221 } else {
222 regex_str += ch;
223 }
224 }
225
226 std::regex re(regex_str, std::regex::icase);
227 return std::regex_match(filename, re);
228}

284std::optional<std::string>
285apply_str_replace(const std::string& content, const std::string& old_str, const std::string& new_str, bool replace_all, std::string& error_type) {
286
287 int occurrences = count_occurrences(content, old_str);
288 if (occurrences == 0) {
289 error_type = "not_found";
290 return std::nullopt;
291 }
292 if (!replace_all && occurrences > 1) {
293 error_type = "multiple_matches";
294 return std::nullopt;
295 }
296
297 std::string result = content;
298 auto pos = result.find(old_str);
299 while (pos != std::string::npos) {
300 result.replace(pos, old_str.size(), new_str);
301 if (!replace_all) { break; }
302 pos = result.find(old_str, pos + new_str.size());
303 }
304 return result;
305}

316std::string apply_insert(const std::string& content,
317 int line_num,
318 const std::string& new_str) {
319 std::istringstream stream(content);
320 std::ostringstream out;
321 std::string line;
322 int current = 0;
323
324 while (std::getline(stream, line)) {
325 ++current;
326 if (current == line_num) {
327 out << new_str << '\n';
328 }
329 out << line << '\n';
330 }
331
332 if (line_num > current) {
333 out << new_str << '\n';
334 }
335 return out.str();

347std::vector<std::string> collect_glob_matches(
348 const fs::path& root,
349 const std::string& pattern,
350 int max_results) {
351
352 std::vector<std::string> matches;
353 auto it = fs::recursive_directory_iterator(
354 root, fs::directory_options::skip_permission_denied);
355
356 for (auto& entry : it) {
357 if (static_cast<int>(matches.size()) >= max_results) {
358 break;
359 }
360 if (entry.is_directory() &&
361 should_skip_dir(
362 entry.path().filename().string())) {
363 it.disable_recursion_pending();
364 continue;
365 }
366 if (!entry.is_regular_file()) {
367 continue;
368 }
369 auto fname = entry.path().filename().string();
370 if (glob_match(fname, pattern)) {
371 matches.push_back(entry.path().string());
372 }
373 }
374 return matches;
375}

386void grep_file(const fs::path& path,
387 const std::regex& re,
388 std::vector<json>& matches,
389 int limit) {
390 std::ifstream in(path);
391 if (!in.is_open()) {
392 return;
393 }
394
395 std::string line;
396 int line_num = 0;
397 while (std::getline(in, line)) {
398 ++line_num;
399 if (static_cast<int>(matches.size()) >= limit) {
400 return;
401 }
402 if (!std::regex_search(line, re)) {
403 continue;
404 }
405 json m;
406 m["path"] = path.string();
407 m["line"] = line_num;
408 m["content"] = line;
409 matches.push_back(std::move(m));
410 }

445std::vector<json> collect_entries(const fs::path& dir,
446 bool recursive,
447 int max_depth) {
448 std::vector<json> entries;
449
450 if (!recursive) {
451 for (auto& entry : fs::directory_iterator(dir)) {
452 entries.push_back(entry_to_json(entry));
453 }
454 return entries;
455 }
456
457 auto it = fs::recursive_directory_iterator(
458 dir, fs::directory_options::skip_permission_denied);
459 for (auto& entry : it) {
460 if (it.depth() > max_depth) {
461 it.disable_recursion_pending();
462 continue;

464 entries.push_back(entry_to_json(entry));
465 }

621ServerResponse ReadFileTool::execute(const std::string& args_json) {
622 auto args = json::parse(args_json);
623 auto requested = args.at("path").get<std::string>();
624 auto resolved = server_.resolve_path(requested);
625 auto path_str = resolved.string();
626
627 if (!fs::exists(resolved)) {
628 return {make_error("not_found",
629 "File not found: " + path_str), {}};
630 }
631
632 auto size = static_cast<int>(fs::file_size(resolved));
633 int limit = server_.max_read_bytes();
634 if (limit > 0 && size > limit) {
635 return {make_error("size_exceeded",
636 "File " + path_str + " is " +
637 std::to_string(size) + " bytes (limit: " +
638 std::to_string(limit) + ")"), {}};
639 }
640
641 auto content = read_file_contents(resolved);
642 server_.tracker().record_read(path_str,
643 hash_content(content));
644
645 logger->info("Read file: {} ({} bytes)", path_str, size);
646 return {build_read_result(path_str, content), {}};
647}

885ServerResponse GrepTool::execute(const std::string& args_json) {
886 auto args = json::parse(args_json);
887 auto pattern = args.at("pattern").get<std::string>();
888 auto file_glob = args.value("glob", std::string("*"));
889
890 std::regex re(pattern);
891 constexpr int MAX_GREP_RESULTS = 100;
892 std::vector<json> matches;
893
894 auto root = fs::current_path();
895 auto it = fs::recursive_directory_iterator(
896 root, fs::directory_options::skip_permission_denied);
897
898 for (auto& entry : it) {
899 if (static_cast<int>(matches.size()) >=
900 MAX_GREP_RESULTS) {
901 break;
902 }
903 if (entry.is_directory() &&
904 should_skip_dir(
905 entry.path().filename().string())) {
906 it.disable_recursion_pending();
907 continue;
908 }
909 if (!entry.is_regular_file()) {
910 continue;
911 }
912 auto fname = entry.path().filename().string();
913 if (!glob_match(fname, file_glob)) {
914 continue;
915 }
916 grep_file(entry.path(), re, matches,
917 MAX_GREP_RESULTS);
918 }
919
920 logger->info("Grep '{}': {} matches", pattern,
921 matches.size());
922 json result = matches;
923 return {result.dump(), {}};
