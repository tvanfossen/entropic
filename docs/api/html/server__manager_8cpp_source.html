69std::string ServerManager::list_tools() const {
70 auto all = nlohmann::json::array();
71
72 // In-process servers
73 for (const auto& [name, server] : servers_) {
74 auto tools_json = server->list_tools();
75 auto tools = nlohmann::json::parse(tools_json);
76 for (auto& tool : tools) {
77 std::string orig_name = tool["name"];
78 tool["name"] = name + "." + orig_name;
79 all.push_back(std::move(tool));
80 }
81 }
82
83 // External servers (tools already prefixed by ExternalMCPClient)
84 for (const auto& [name, client] : external_clients_) {
85 if (!client->is_connected()) {
86 continue;
87 }
88 auto tools_json = client->list_tools();
89 auto tools = nlohmann::json::parse(tools_json);
90 for (auto& tool : tools) {
91 all.push_back(std::move(tool));
92 }
93 }
94
95 logger->info("Tool list: {} tools from {} server(s) + {} external",
96 all.size(), servers_.size(),
97 external_clients_.size());
98 return all.dump();
99}

134std::string ServerManager::route_tool_call(
135 const std::string& tool_name,
136 const std::string& args_json) {
137
138 auto prefix = extract_prefix(tool_name);
139 auto local_name = extract_local_name(tool_name);
140
141 // Try in-process server first
142 auto it = servers_.find(prefix);
143 if (it != servers_.end()) {
144 return it->second->execute(local_name, args_json);
145 }
146
147 // Try external client
148 auto ext_it = external_clients_.find(prefix);
149 if (ext_it != external_clients_.end()) {
150 return route_external_call(
151 ext_it->second.get(), tool_name, local_name, args_json);
152 }
153
154 logger->warn("Unknown server: {}", prefix);
155 nlohmann::json resp;
156 resp["result"] = "Error: Unknown server '" + prefix + "'";
157 resp["directives"] = nlohmann::json::array();
158 return resp.dump();
159}

332std::string ServerManager::args_to_pattern(
333 const std::string& args_json) {
334 if (args_json.empty() || args_json == "{}") {
335 return "*";
336 }
337 try {
338 auto j = nlohmann::json::parse(args_json);
339 if (j.is_object() && !j.empty()) {
340 auto first = j.begin();
341 if (first->is_string()) {
342 return first->get<std::string>();
343 }
344 }
345 } catch (...) {
346 // Parse failure — treat as wildcard
347 }
348 return "*";
349}

368void ServerManager::initialize_external_servers() {
369 // Create .mcp.json discovery
370 mcp_json_discovery_ = std::make_unique<MCPJsonDiscovery>(
371 project_dir_);
372
373 // YAML config external_servers
374 for (const auto& [name, entry] : mcp_config_.external_servers) {
375 auto client = create_external_client(name, entry);
376 connect_and_register_external(name, std::move(client),
377 "config", entry.url,
378 entry.command);
379 }
380
381 // .mcp.json discovery
382 std::set<std::string> existing;
383 for (const auto& [name, _] : servers_) {
384 existing.insert(name);
385 }
386 for (const auto& [name, _] : external_clients_) {
387 existing.insert(name);
388 }
389
390 auto discovered = mcp_json_discovery_->discover(existing);
391 for (const auto& cfg : discovered) {
392 auto client = create_external_client(cfg);
393 connect_and_register_external(cfg.name, std::move(client),
394 "mcp_json", cfg.url,
395 cfg.command);
396 }
397
398 // Start health monitor
399 health_monitor_ = std::make_unique<HealthMonitor>(
400 ReconnectPolicy(mcp_config_.reconnect),
401 mcp_config_.health_check_interval_ms);
402
403 for (auto& [name, client] : external_clients_) {
404 health_monitor_->watch(name, client.get());
405 }
406 if (!external_clients_.empty()) {
407 health_monitor_->start();
408 }
409
410 logger->info("External MCP: {} servers connected",
411 external_clients_.size());
412}

424void ServerManager::connect_and_register_external(
425 const std::string& name,
426 std::unique_ptr<ExternalMCPClient> client,
427 const std::string& source,
428 const std::string& url,
429 const std::string& command) {
430
431 ServerInfo info;
432 info.name = name;
433 info.transport = url.empty() ? "stdio" : "sse";
434 info.url = url;
435 info.command = command;
436 info.source = source;
437 info.status = "disconnected";

449 external_clients_[name] = std::move(client);

462std::vector<std::string> ServerManager::connect_external_server(
463 const std::string& name,
464 const std::string& command,
465 const std::vector<std::string>& args,
466 const std::string& url) {
467
468 if (servers_.count(name) > 0 ||
469 external_clients_.count(name) > 0) {
470 logger->warn("Server '{}' already registered", name);
471 return {};
472 }
473
474 std::unique_ptr<Transport> transport;
475 if (!url.empty()) {
476 transport = std::make_unique<SSETransport>(url);
477 } else {
478 transport = std::make_unique<StdioTransport>(
479 command, args);
480 }
481
482 auto client = std::make_unique<ExternalMCPClient>(
483 name, std::move(transport));
484
485 connect_and_register_external(name, std::move(client),
486 "runtime", url, command);
487
488 auto& registered = external_clients_[name];
489 if (health_monitor_) {
490 health_monitor_->watch(name, registered.get());
491 }
492
493 // Parse tool names from cached list
494 std::vector<std::string> tool_names;
495 auto tools_json = registered->list_tools();
496 try {
497 auto tools = nlohmann::json::parse(tools_json);
498 for (const auto& t : tools) {
499 tool_names.push_back(t["name"].get<std::string>());
500 }
501 } catch (...) {}
502
503 return tool_names;
504}

520
