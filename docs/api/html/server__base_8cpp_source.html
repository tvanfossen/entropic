184std::string MCPServerBase::serialize_response(
185 const ServerResponse& response) {
186 nlohmann::json j;
187 j["result"] = response.result;
188
189 auto directives = nlohmann::json::array();
190 for (const auto& d : response.directives) {
191 nlohmann::json dj;
192 dj["type"] = directive_type_name(d.type);
193 directives.push_back(std::move(dj));
194 }
195 j["directives"] = std::move(directives);
196
197 return j.dump();
198}

208void MCPServerBase::inject_anchor_if_needed(
209 const ToolBase& tool,
210 const std::string& args_json,
211 ServerResponse& response) {
212 auto key = tool.anchor_key(args_json);
213 if (key.empty()) {

216 Directive anchor;
217 anchor.type = ENTROPIC_DIRECTIVE_CONTEXT_ANCHOR;
218 response.directives.push_back(std::move(anchor));
