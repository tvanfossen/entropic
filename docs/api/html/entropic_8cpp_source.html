119static entropic_error_t configure_common(entropic_handle_t h) {
120 auto data_dir = entropic::config::resolve_data_dir(h->config);
121 auto models_yaml = data_dir / "bundled_models.yaml";
122
123 auto err = h->bundled_models.load(models_yaml);
124 if (!err.empty()) {
125 h->last_error = "bundled models: " + err;
126 s_log->error("{}", h->last_error);
127 return ENTROPIC_ERROR_INVALID_CONFIG;
128 }
129
130 h->orchestrator = std::make_unique<entropic::ModelOrchestrator>();
131 if (!h->orchestrator->initialize(h->config)) {
132 h->last_error = "orchestrator initialization failed";
133 s_log->error("{}", h->last_error);
134 return ENTROPIC_ERROR_LOAD_FAILED;
135 }
136
137 h->mcp_auth = std::make_unique<entropic::MCPAuthorizationManager>();
138 h->identity_manager = std::make_unique<entropic::IdentityManager>(
139 entropic::IdentityManagerConfig{});
140 h->server_manager = std::make_unique<entropic::ServerManager>(
141 h->config.permissions, std::filesystem::current_path());
142
143 h->configured.store(true);
144 s_log->info("configure complete — data_dir={}", data_dir.string());
145 return ENTROPIC_OK;
146}

269entropic_error_t entropic_run(

271 const char* input,

273 REQUIRE_ORCHESTRATOR(handle);

276 try {
277 entropic::Message msg;
278 msg.role = "user";
279 msg.content = input;
280 auto result = handle->engine->run({msg});
281 nlohmann::json j = nlohmann::json::array();
282 for (const auto& m : result) {
283 j.push_back({{"role", m.role}, {"content", m.content}});
284 }
285 *result_json = strdup(j.dump().c_str());
286 return ENTROPIC_OK;
287 } catch (const std::exception& e) {
288 handle->last_error = e.what();
289 s_log->error("run: {}", handle->last_error);
290 return ENTROPIC_ERROR_GENERATE_FAILED;

301entropic_error_t entropic_run_streaming(

304 void (*on_token)(const char* token, size_t len, void* user_data),
305 void* user_data,
306 int* cancel_flag) {
307 REQUIRE_ORCHESTRATOR(handle);
308 if (!input || !on_token) { return ENTROPIC_ERROR_INVALID_ARGUMENT; }
309 try {
310 std::atomic<bool> cancel{false};
311 auto token_cb = [&](std::string_view tok) {
312 on_token(tok.data(), tok.size(), user_data);

315 entropic::Message msg;
316 msg.role = "user";
317 msg.content = input;
318 auto result = handle->orchestrator->generate_streaming(
319 {msg}, entropic::GenerationParams{}, token_cb, cancel);
320 return ENTROPIC_OK;
321 } catch (const std::exception& e) {
322 handle->last_error = e.what();
323 s_log->error("run_streaming: {}", handle->last_error);
324 return ENTROPIC_ERROR_GENERATE_FAILED;

359 const char* adapter_path,
360 const char* base_model_path,
361 float scale)
362{
363 REQUIRE_ORCHESTRATOR(handle);
364 if (!adapter_name || !adapter_path || !base_model_path) {

371 auto* base = handle->orchestrator->get_backend(tier);
372 auto* llama = dynamic_cast<entropic::LlamaCppBackend*>(base);
373 if (!llama || !llama->llama_model_ptr()) {
374 return ENTROPIC_ERROR_INVALID_STATE;
375 }
376 bool ok = handle->orchestrator->adapter_manager().load(
377 adapter_name, adapter_path, llama->llama_model_ptr(), scale);
378 return ok ? ENTROPIC_OK : ENTROPIC_ERROR_ADAPTER_LOAD_FAILED;
379 } catch (const std::exception& e) {

1216entropic_error_t entropic_get_logprobs(
1217 entropic_handle_t handle,
1218 const char* model_id,
1219 const int32_t* tokens,
1220 int n_tokens,
1221 entropic_logprob_result_t* result)
1222{
1223 REQUIRE_ORCHESTRATOR(handle);
1224 if (!model_id || !tokens || !result || n_tokens < 2) {
1225 return ENTROPIC_ERROR_INVALID_ARGUMENT;
1226 }
1227 try {
1228 std::lock_guard lock(handle->api_mutex);
1229 auto* backend = require_active_backend(handle, model_id);
1230 auto lr = backend->evaluate_logprobs(tokens, n_tokens);
1231 result->n_tokens = lr.n_tokens;
1232 result->n_logprobs = lr.n_logprobs;
1233 result->perplexity = lr.perplexity;
1234 result->total_logprob = lr.total_logprob;
1235 result->logprobs = static_cast<float*>(
1236 malloc(sizeof(float) * lr.logprobs.size()));
1237 std::copy(lr.logprobs.begin(), lr.logprobs.end(),
1238 result->logprobs);
1239 result->tokens = static_cast<int32_t*>(
1240 malloc(sizeof(int32_t) * lr.tokens.size()));
1241 std::copy(lr.tokens.begin(), lr.tokens.end(),
1242 result->tokens);
1243 return ENTROPIC_OK;
1244 } catch (const std::exception& e) {
1245 handle->last_error = e.what();
1246 s_log->error("get_logprobs: {}", handle->last_error);
1247 return ENTROPIC_ERROR_EVAL_FAILED;
1248 }
