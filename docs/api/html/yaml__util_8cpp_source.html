38std::string read_file(const std::filesystem::path& path)
39{
40 std::ifstream ifs(path);
41 if (!ifs.is_open()) {
42 s_log->error("Cannot open file: {}", path.string());
43 return "";
44 }
45 std::ostringstream ss;
46 ss << ifs.rdbuf();
47 auto content = ss.str();
48 s_log->info("Read file: {}, {} bytes", path.string(),
49 content.size());
50 return content;
51}

60std::filesystem::path expand_home(const std::filesystem::path& p)
61{
62 auto str = p.string();
63 const char* home = std::getenv("HOME");
64 bool needs_expansion = !str.empty() && str[0] == '~' && home != nullptr;
65
66 if (!needs_expansion) {
67 return p;
68 }
69
70 return (str.size() == 1)
71 ? std::filesystem::path(home)
72 : std::filesystem::path(home) / str.substr(2);
