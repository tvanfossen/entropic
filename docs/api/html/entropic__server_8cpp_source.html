132ServerResponse TodoTool::execute(const std::string& args_json) {
133 auto args = nlohmann::json::parse(args_json);
134 std::string action = args.at("action").get<std::string>();
135
136 if (action == "add") {
137 std::string content = args.at("content").get<std::string>();
138 items_.push_back({content, "pending"});
139 logger->info("[todo] add: {}", content);
140 } else if (action == "update") {
141 auto idx = args.at("index").get<size_t>();
142 if (idx < items_.size()) {
143 items_[idx].status = args.value("status", items_[idx].status);
144 items_[idx].content = args.value("content", items_[idx].content);
145 logger->info("[todo] update #{}: {}", idx, items_[idx].status);
146 }
147 } else if (action == "remove") {
148 auto idx = args.at("index").get<size_t>();
149 if (idx < items_.size()) {
150 logger->info("[todo] remove #{}", idx);
151 items_.erase(items_.begin() +
152 static_cast<ptrdiff_t>(idx));
153 }
154 }
155
156 std::string list_text = format_list();
157
158 nlohmann::json result;
159 result["todo_state"] = list_text;
160 result["action"] = action;
161
162 Directive anchor_d;
163 anchor_d.type = ENTROPIC_DIRECTIVE_CONTEXT_ANCHOR;
164
165 Directive notify_d;
166 notify_d.type = ENTROPIC_DIRECTIVE_NOTIFY_PRESENTER;
167
168 return {result.dump(), {anchor_d, notify_d}};
169}

627static nlohmann::json build_snapshot(
628 const entropic_state_provider_t& p,
629 bool include_docs, int history_limit) {
630
631 nlohmann::json snap;
632
633 auto now = std::chrono::system_clock::now();
634 auto ms = std::chrono::duration_cast<
635 std::chrono::milliseconds>(
636 now.time_since_epoch()).count();
637 snap["snapshot_timestamp_ms"] = ms;
638
639 snap["engine"] = nlohmann::json::parse(
640 call_provider(p.get_state, p.user_data));
641 snap["config"] = nlohmann::json::parse(
642 call_provider(p.get_config, p.user_data));
643 snap["identities"] = nlohmann::json::parse(
644 call_provider(p.get_identities, p.user_data));
645 snap["tools"] = nlohmann::json::parse(
646 call_provider(p.get_tools, p.user_data));
647 snap["history"] = nlohmann::json::parse(
648 call_history_provider(
649 p.get_history, history_limit, p.user_data));
650 snap["metrics"] = nlohmann::json::parse(
651 call_provider(p.get_metrics, p.user_data));
652
653 if (include_docs) {
654 snap["docs"] = call_docs_provider(
655 p.get_docs, nullptr, p.user_data);
656 } else {
657 snap["docs"] = nullptr;
658 }
659 return snap;

797static std::string filter_json_by_key(
798 const std::string& json_str,
799 const std::string& key,
800 const std::string& label) {
801 auto j = nlohmann::json::parse(json_str);
802
803 if (j.is_object() && j.contains(key)) {
804 return j[key].dump();
805 }
806 if (j.is_array()) {
807 for (const auto& item : j) {
808 if (item.value("name", "") == key) {
809 return item.dump();
810 }
811 }
812 }
813 return "Error: " + label + " '" + key
814 + "' not found. Available: "
815 + list_available_keys(j);
816}

848static bool dispatch_simple_target(
849 const entropic_state_provider_t& p,
850 const std::string& target,
851 const std::string& key,
852 std::string& result) {
853
854 if (target == "state") {
855 result = call_provider(p.get_state, p.user_data);
856 } else if (target == "metrics") {
857 result = call_provider(p.get_metrics, p.user_data);
858 } else if (target == "history") {
859 int limit = key.empty() ? 10 : std::atoi(key.c_str());
860 result = call_history_provider(
861 p.get_history, limit, p.user_data);
862 } else if (target == "docs") {
863 const char* sec = key.empty() ? nullptr : key.c_str();
864 result = call_docs_provider(p.get_docs, sec, p.user_data);
865 } else {
866 return false;
867 }
868 return true;
869}

881static bool dispatch_filterable_target(
882 const entropic_state_provider_t& p,
883 const std::string& target,
884 const std::string& key,
885 std::string& result) {

889 } else if (target == "identity") {
890 result = inspect_filterable(

892 } else if (target == "tool") {
893 result = inspect_filterable(
894 p.get_tools, p.user_data, key, "tool");
895 } else {
896 return false;
897 }

910static std::string dispatch_inspect(
911 const entropic_state_provider_t& p,
912 const std::string& target,
913 const std::string& key) {
914
915 std::string result;
916 if (dispatch_filterable_target(p, target, key, result)) {
917 return result;
918 }
919 if (dispatch_simple_target(p, target, key, result)) {
920 return result;
921 }
922 return "Error: unknown target '" + target
923 + "'. Supported: config, identity, tool, state, "
