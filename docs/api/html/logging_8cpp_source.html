79void escape_ansi(const std::string& text, size_t& i,
80 std::string& result)
81{
82 result += "[ESC";
83 if (i + 1 >= text.size() || text[i + 1] != '[') {
84 result += ']';
85 return;
86 }
87 ++i; // skip '['
88 while (i + 1 < text.size() && text[i + 1] != 'm') {
89 ++i;
90 result += text[i];
91 }
92 if (i + 1 < text.size()) { ++i; } // skip 'm'
93 result += ']';
94}

142std::string escape_content(const std::string& text) {
143 std::string result;
144 result.reserve(text.size() + text.size() / 8);
145 for (size_t i = 0; i < text.size(); ++i) {
146 if (escape_char(text[i], result)) { continue; }
147 if (text[i] == '\033') { escape_ansi(text, i, result); }
148 else { result += text[i]; }
149 }
150 return result;
151}
