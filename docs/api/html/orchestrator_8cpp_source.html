58bool ModelOrchestrator::initialize(const ParsedConfig& config) {
59 config_ = config;
60 default_tier_ = config.models.default_tier;
61
62 logger->info("Initializing model orchestrator");
63
64 // Create deduplicated backends per unique model path
65 for (const auto& [name, tier_config] : config.models.tiers) {
66 std::string path_key = tier_config.path.string();
67
68 if (model_pool_.find(path_key) == model_pool_.end()) {
69 auto backend = std::make_shared<LlamaCppBackend>();
70 model_pool_[path_key] = backend;
71 }
72 tiers_[name] = model_pool_[path_key];
73
74 // Per-tier adapter (identity-specific)
75 adapters_[name] = create_adapter(
76 tier_config.adapter, name, "" /* prompt resolved later */);
77 }
78
79 logger->info("Created {} unique backend(s) for {} tier(s)",
80 model_pool_.size(), tiers_.size());
81
82 // Router backend (separate small model)
83 if (config.models.router) {
84 router_ = std::make_shared<LlamaCppBackend>();
85 }
86
87 // Build routing data
88 for (const auto& [digit, tier_name] : config.routing.tier_map) {
89 tier_map_[digit] = tier_name;
90 }
91 for (const auto& [src, targets] : config.routing.handoff_rules) {
92 handoff_rules_[src] = std::unordered_set<std::string>(
93 targets.begin(), targets.end());
94 }
95
96 // Load default tier
97 if (tiers_.find(default_tier_) != tiers_.end()) {
98 auto& backend = tiers_[default_tier_];
99 auto& tier_cfg = config.models.tiers.at(default_tier_);
100 if (backend->load_and_activate(tier_cfg)) {
101 loaded_main_tier_ = default_tier_;
102 logger->info("Activated default tier: {}", default_tier_);
103 } else {
104 logger->error("Failed to activate default tier: {}", default_tier_);
105 return false;
106 }
107 }
108
109 // Load router
110 if (router_ && config.models.router) {
111 if (!router_->load_and_activate(*config.models.router)) {
112 logger->error("Failed to activate router model");
113 } else {
114 logger->info("Activated router model");
115 }
116 }
117
118 // Preload LoRA adapters to WARM (v1.9.2)
119 preload_adapters();
120
121 // Load bundled grammars (v1.9.3)
122 load_bundled_grammars();
123
124 return true;
125}

157GenerationResult ModelOrchestrator::generate(
158 const std::vector<Message>& messages,
159 const GenerationParams& params,
160 const std::string& tier_name)
161{
162 auto t_start = now();
163
164 // Route if no explicit tier
165 std::string selected = tier_name;
166 double routing_ms = 0.0;
167 if (selected.empty()) {
168 auto t_route = now();
169 selected = route(messages);
170 routing_ms = elapsed_ms(t_route, now());
171 }
172
173 // Get model (may trigger swap)
174 auto t_swap = now();
175 InferenceBackend* model = get_model(selected);
176 double swap_ms = elapsed_ms(t_swap, now());
177
178 if (!model) {
179 GenerationResult err;
180 err.error_code = ENTROPIC_ERROR_GENERATE_FAILED;
181 err.error_message = "No model available for tier: " + selected;
182 err.finish_reason = "error";
183 return err;
184 }
185
186 // Resolve grammar_key → grammar content (v1.9.3)
187 GenerationParams resolved_params = params;
188 resolve_grammar_key(resolved_params, selected);
189
190 // Generate
191 auto result = model->generate(messages, resolved_params);
192
193 // Parse tool calls via adapter
194 auto* adapter = get_adapter(selected);
195 if (adapter && !result.content.empty()) {
196 result.raw_content = result.content;
197 auto parsed = adapter->parse_tool_calls(result.content);
198 result.content = parsed.cleaned_content;
199 result.tool_calls = std::move(parsed.tool_calls);
200 }
201
202 result.routing_ms = routing_ms;
203 result.swap_ms = swap_ms;
204 result.total_ms = elapsed_ms(t_start, now());
205 logger->info("Orchestration: tier={}, adapter={}, grammar={}",
206 selected, last_routing_result_.adapter_name,
207 resolved_params.grammar.empty()
208 ? "unconstrained" : resolved_params.grammar_key);
209 logger->info("Total: {:.0f}ms (route={:.0f}ms, swap={:.0f}ms, "
210 "gen={:.0f}ms)",
211 result.total_ms, routing_ms, swap_ms,
212 result.generation_time_ms);
213 return result;
214}

255std::string ModelOrchestrator::route(const std::vector<Message>& messages) {
256 if (!config_.routing.enabled || !router_) {
257 logger->info("Route: routing disabled, using default '{}'",
258 default_tier_);
259 last_routing_result_ = {default_tier_, "", "", "none", 0.0};
260 return default_tier_;
261 }
262
263 auto [tier, raw] = classify_task(messages);
264 last_routing_result_ = {tier, loaded_main_tier_, raw, "none", 0.0};
265
266 // Track history
267 tier_history_.push_back(tier);
268 if (tier_history_.size() > 5) {
269 tier_history_.erase(tier_history_.begin());
270 }
271
272 logger->info("[ROUTER] {} | raw='{}'", tier, raw);
273 return tier;
274}

283std::pair<std::string, std::string> ModelOrchestrator::classify_task(
284 const std::vector<Message>& messages)
285{
286 std::string user_msg = extract_latest_user_message(messages);
287
288 GenerationParams router_params;
289 router_params.max_tokens = 1;
290 router_params.temperature = 0.0f;
291
292 auto result = router_->complete(user_msg + " ->", router_params);
293 std::string raw = result.content;
294
295 // Trim whitespace
296 auto start = raw.find_first_not_of(" \t\n\r");
297 if (start != std::string::npos) {
298 raw = raw.substr(start);
299 }
300
301 // Find matching tier
302 for (char c : raw) {
303 std::string digit(1, c);
304 auto it = tier_map_.find(digit);
305 if (it != tier_map_.end()) {
306 logger->info("Route: digit='{}' -> tier='{}'",
307 digit, it->second);
308 return {it->second, digit};
309 }
310 }
311
312 logger->warn("Route: no valid digit in '{}', defaulting to {}",
313 raw, default_tier_);
314 return {default_tier_, ""};
315}

326InferenceBackend* ModelOrchestrator::get_model(const std::string& tier_name) {
327 std::lock_guard<std::mutex> lock(swap_mutex_);
328
329 auto it = tiers_.find(tier_name);
330 if (it == tiers_.end()) {
331 it = tiers_.find(config_.routing.fallback_tier);
332 }
333
334 InferenceBackend* result = nullptr;
335
336 if (it == tiers_.end()) {
337 /* no tier found — result stays nullptr */
338 } else if (it->second->is_active()) {
339 last_routing_result_.swap_action = "reused";
340 result = it->second.get();
341 } else {
342 auto& backend = it->second;
343 deactivate_current_if_needed(backend.get());
344
345 auto tier_it = config_.models.tiers.find(tier_name);
346 bool activated = tier_it != config_.models.tiers.end()
347 && backend->load_and_activate(tier_it->second);
348
349 if (activated) {
350 loaded_main_tier_ = tier_name;
351 last_routing_result_.swap_action = "loaded";
352 result = backend.get();
353 } else {
354 logger->error("Failed to activate tier: {}", tier_name);
355 }
356 }
357
358 // Ensure correct LoRA adapter for this tier (v1.9.2)
359 if (result) {
360 auto* llama_backend = dynamic_cast<LlamaCppBackend*>(result);
361 llama_context* ctx = llama_backend
362 ? llama_backend->llama_context_ptr() : nullptr;
363 double adapter_ms = ensure_adapter_for_tier(tier_name, ctx);
364 last_routing_result_.adapter_swap_ms = adapter_ms;
365 last_routing_result_.adapter_name = lora_manager_.active_adapter();
366 }
367
368 return result;
369}

380void ModelOrchestrator::deactivate_current_if_needed(InferenceBackend* incoming) {
381 auto it = loaded_main_tier_.empty()
382 ? tiers_.end() : tiers_.find(loaded_main_tier_);
383
384 bool should_swap = it != tiers_.end()
385 && it->second.get() != incoming
386 && it->second->is_loaded();
387
388 if (!should_swap) {
389 return;
390 }
391
392 auto& current = it->second;
393 auto cfg_it = config_.models.tiers.find(loaded_main_tier_);
394 bool keep_warm = cfg_it != config_.models.tiers.end() && cfg_it->second.keep_warm;
395
396 // Cascade: unload adapters for this base model (v1.9.2)
397 auto* llama_backend = dynamic_cast<LlamaCppBackend*>(current.get());
398 if (llama_backend) {
399 lora_manager_.unload_all_for_model(
400 llama_backend->llama_model_ptr(),
401 llama_backend->llama_context_ptr());
402 }
403
404 if (keep_warm) {
405 logger->info("Deactivating {} (keep_warm=true)", loaded_main_tier_);
406 current->deactivate();
407 } else {
408 logger->info("Unloading {} (keep_warm=false)", loaded_main_tier_);
409 current->unload();
410 }
411}

547double ModelOrchestrator::ensure_adapter_for_tier(
548 const std::string& tier_name, llama_context* ctx)
549{
550 auto tier_it = config_.models.tiers.find(tier_name);
551 if (tier_it == config_.models.tiers.end()) {
552 return 0.0;
553 }
554
555 const auto& tier_cfg = tier_it->second;
556 auto t_start = now();
557 bool needs_kv_clear = false;
558
559 if (!tier_cfg.adapter_path) {
560 needs_kv_clear = deactivate_if_active(ctx);
561 } else if (lora_manager_.active_adapter() != tier_name) {
562 needs_kv_clear = lora_manager_.swap(tier_name, ctx);
563 if (!needs_kv_clear) {
564 logger->warn("Adapter swap to '{}' failed", tier_name);
565 }
566 }
567
568 if (needs_kv_clear && ctx) {
569 llama_memory_clear(llama_get_memory(ctx), true);
570 logger->info("Adapter swap for tier '{}' in {:.1f}ms",
571 tier_name, elapsed_ms(t_start, now()));
572 }
573
574 return elapsed_ms(t_start, now());
575}

586void ModelOrchestrator::preload_adapters() {
587 int loaded = 0;
588
589 for (const auto& [name, tier_cfg] : config_.models.tiers) {
590 if (!tier_cfg.adapter_path) {
591 continue;
592 }
593
594 auto tier_it = tiers_.find(name);
595 if (tier_it == tiers_.end()) {
596 continue;
597 }
598
599 auto* llama_backend = dynamic_cast<LlamaCppBackend*>(
600 tier_it->second.get());
601 if (!llama_backend || !llama_backend->llama_model_ptr()) {
602 logger->warn("Cannot preload adapter for '{}' — model not loaded",
603 name);
604 continue;
605 }
606
607 bool ok = lora_manager_.load(
608 name,
609 *tier_cfg.adapter_path,
610 llama_backend->llama_model_ptr(),
611 tier_cfg.adapter_scale);
612
613 if (ok) {
614 ++loaded;
615 }
616 }
617
618 if (loaded > 0) {
619 logger->info("Preloaded {} LoRA adapter(s) to WARM", loaded);
620 }
621}

682void ModelOrchestrator::resolve_grammar_key(
683 GenerationParams& params, const std::string& tier_name)
684{
685 if (!params.grammar.empty()) {
686 return;
687 }
688
689 // Try explicit grammar_key
690 std::string key = params.grammar_key;
691
692 // Fall back to tier config grammar field (frontmatter)
693 if (key.empty()) {
694 auto it = config_.models.tiers.find(tier_name);
695 if (it != config_.models.tiers.end() && it->second.grammar) {
696 key = normalize_grammar_key(it->second.grammar->string());
697 }
698 }
699
700 if (key.empty()) {
701 return;
702 }
703
704 std::string content = grammar_registry_.get(key);
705 if (content.empty()) {
706 logger->warn("Grammar key '{}' not found in registry", key);
707 return;
708 }
709
710 logger->info("Grammar resolved: key='{}', {} bytes",
711 key, content.size());
712 params.grammar = std::move(content);
713}
