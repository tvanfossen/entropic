92static entropic_error_t check_create_preconditions(
93 const IdentityConfig& config,
94 const IdentityManagerConfig& mgr_config,
95 const std::unordered_map<std::string, IdentityConfig>& identities) {
96 if (!mgr_config.allow_dynamic) {
97 logger->warn("Dynamic identity creation disabled");
98 return ENTROPIC_ERROR_INVALID_CONFIG;
99 }
100 if (identities.size() >= mgr_config.max_identities) {
101 logger->warn("Identity limit reached: {}",
102 mgr_config.max_identities);
103 return ENTROPIC_ERROR_LIMIT_REACHED;
104 }
105 bool exists = identities.count(config.name) > 0;
106 if (exists) {
107 logger->warn("Identity already exists: {}", config.name);
108 }
109 return exists ? ENTROPIC_ERROR_ALREADY_EXISTS : ENTROPIC_OK;
110}

194// ── create ───────────────────────────────────────────────
195
203entropic_error_t IdentityManager::create(const IdentityConfig& config) {
204 std::unique_lock lock(identities_mutex_);
205 auto pre = check_create_preconditions(
206 config, config_, identities_);
207 if (pre != ENTROPIC_OK) { return pre; }
208 auto err = validate(config, false);
209 if (!err.empty()) {
210 logger->warn("Identity validation failed: {}", err);
211 return ENTROPIC_ERROR_INVALID_CONFIG;
212 }
213 IdentityConfig cfg = config;
214 cfg.origin = IdentityOrigin::DYNAMIC;
215 register_mcp_keys(cfg);
216 identities_[cfg.name] = std::move(cfg);
217 router_dirty_.store(true, std::memory_order_release);
218 logger->info("Identity created: name='{}', routable={}, "
219 "prompt={} chars",
220 config.name, config.routable,
221 config.system_prompt.size());
222 return ENTROPIC_OK;
223}

225// ── update ───────────────────────────────────────────────
226
235entropic_error_t IdentityManager::update(
236 const std::string& name,
237 const IdentityConfig& config) {
238 std::unique_lock lock(identities_mutex_);
239 std::unordered_map<std::string,
240 IdentityConfig>::const_iterator it;
241 auto pre = check_mutable(name, identities_, it);
242 if (pre != ENTROPIC_OK) { return pre; }
243 auto err = validate(config, true);
244 if (!err.empty()) {
245 logger->warn("Identity update validation failed: {}", err);
246 return ENTROPIC_ERROR_INVALID_CONFIG;
247 }
248 unregister_mcp_keys(name);
249 IdentityConfig cfg = config;
250 cfg.origin = IdentityOrigin::DYNAMIC;
251 cfg.name = name;
252 register_mcp_keys(cfg);
253 identities_[name] = std::move(cfg);
254 router_dirty_.store(true, std::memory_order_release);
255 logger->info("Dynamic identity updated: {}", name);
256 return ENTROPIC_OK;
257}

259// ── destroy ──────────────────────────────────────────────
260
268entropic_error_t IdentityManager::destroy(const std::string& name) {
269 std::unique_lock lock(identities_mutex_);
270 std::unordered_map<std::string,
271 IdentityConfig>::const_iterator it;
272 auto pre = check_mutable(name, identities_, it);
273 if (pre != ENTROPIC_OK) { return pre; }
274 if (in_use_checker_ && in_use_checker_(name, in_use_user_data_)) {
275 return ENTROPIC_ERROR_IN_USE;
276 }
277 unregister_mcp_keys(name);
278 identities_.erase(it);
279 router_dirty_.store(true, std::memory_order_release);
280 logger->info("Dynamic identity destroyed: {}", name);
281 return ENTROPIC_OK;
282}

428std::string IdentityManager::validate_name(const std::string& name) {
429 if (name.empty()) {
430 return "Identity name cannot be empty";
431 }
432 if (!std::regex_match(name, s_name_regex)) {
433 return "Identity name must match ^[a-z][a-z0-9_-]{0,63}$";
434 }
435 bool reserved = std::find(
436 s_reserved_names.begin(), s_reserved_names.end(), name)
437 != s_reserved_names.end();
438 return reserved
439 ? "Identity name '" + name + "' is reserved" : "";

450std::string IdentityManager::validate(
451 const IdentityConfig& config, bool is_update) const {
452 if (!is_update) {
453 auto name_err = validate_name(config.name);
454 if (!name_err.empty()) { return name_err; }
455 }
456 if (config.focus.empty()) {
457 return "Identity must have at least one focus keyword";
458 }
459 bool grammar_missing = !config.grammar_id.empty()
460 && grammar_iface_.has_grammar
461 && !grammar_iface_.has_grammar(
462 config.grammar_id, grammar_iface_.user_data);
463 return grammar_missing
464 ? "Grammar '" + config.grammar_id + "' not found in registry"
465 : validate_fields(config);
466}

476void IdentityManager::register_mcp_keys(const IdentityConfig& config) {
477 if (config.mcp_keys.empty()) { return; }
478 if (!mcp_iface_.register_identity) { return; }
479 mcp_iface_.register_identity(config.name, mcp_iface_.user_data);
480 for (const auto& key : config.mcp_keys) {
481 if (mcp_iface_.grant) {
482 mcp_iface_.grant(config.name, key.tool_pattern,
483 static_cast<int>(key.level),
484 mcp_iface_.user_data);
485 }
486 }
