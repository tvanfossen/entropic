136void HealthMonitor::monitor_loop() {
137 constexpr auto poll_interval = std::chrono::milliseconds(500);
138
139 while (running_) {
140 {
141 std::lock_guard<std::mutex> lock(watched_mutex_);
142 auto now = std::chrono::steady_clock::now();
143 for (auto& [name, entry] : watched_) {
144 if (now >= entry.next_action) {
145 check_server(name, entry);
146 }
147 }
148 }
149
150 std::unique_lock<std::mutex> lock(wake_mutex_);
151 wake_cv_.wait_for(lock, poll_interval,
152 [this] { return !running_.load(); });
153 }

163void HealthMonitor::check_server(
164 const std::string& name,
165 WatchEntry& entry) {
166
167 bool alive = entry.client->is_connected();
168
169 if (entry.status == "connected" && !alive) {
170 // Connection lost — start reconnecting
171 auto old = entry.status;
172 entry.status = "disconnected";
173 entry.reconnect_attempt = 0;
174 post_event(name, old, entry.status);
175 logger->warn("Server '{}' disconnected", name);
176 attempt_reconnect(name, entry);
177 return;
178 }
179
180 if (entry.status == "disconnected" ||
181 entry.status == "reconnecting") {
182 attempt_reconnect(name, entry);
183 return;
184 }
185
186 // Connected + healthy: schedule next check
187 if (health_check_interval_ms_ > 0) {
188 entry.next_action = std::chrono::steady_clock::now() +
189 std::chrono::milliseconds(health_check_interval_ms_);
190 } else {
191 entry.next_action = std::chrono::steady_clock::now() +
192 std::chrono::seconds(5);
193 }
194}

203void HealthMonitor::attempt_reconnect(
204 const std::string& name,
205 WatchEntry& entry) {
206
207 if (policy_.exhausted(entry.reconnect_attempt)) {
208 if (entry.status != "error") {
209 auto old = entry.status;
210 entry.status = "error";
211 post_event(name, old, "error");
212 logger->error("Server '{}' reconnection exhausted "
213 "after {} attempts",
214 name, entry.reconnect_attempt);
215 }
216 entry.next_action = std::chrono::steady_clock::time_point::max();
217 return;
218 }
219
220 if (entry.status != "reconnecting") {
221 auto old = entry.status;
222 entry.status = "reconnecting";
223 post_event(name, old, "reconnecting");
224 }
225
226 logger->info("Reconnecting to '{}' (attempt {})",
227 name, entry.reconnect_attempt + 1);
228
229 bool ok = entry.client->connect();
230 if (ok) {
231 auto [added, removed] = entry.client->refresh_tools();
232 entry.status = "connected";
233 entry.reconnect_attempt = 0;
234
235 HealthEvent evt;
236 evt.server_name = name;
237 evt.old_status = "reconnecting";
238 evt.new_status = "connected";
239 evt.added_tools = std::move(added);
240 evt.removed_tools = std::move(removed);
241 {
242 std::lock_guard<std::mutex> lock(event_mutex_);
243 event_queue_.push_back(std::move(evt));
244 }
245
246 logger->info("Server '{}' reconnected", name);
247 entry.next_action = std::chrono::steady_clock::now();
248 return;
249 }
250
251 // Schedule next attempt with backoff
252 auto delay = policy_.delay_ms(entry.reconnect_attempt);
253 entry.reconnect_attempt++;
254 entry.next_action = std::chrono::steady_clock::now() +
255 std::chrono::milliseconds(delay);
256
257 logger->info("Server '{}' reconnect failed, "
258 "retry in {}ms", name, delay);
259}
