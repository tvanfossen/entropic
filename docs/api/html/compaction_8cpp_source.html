123CompactionResult CompactionManager::check_and_compact(
124 std::vector<Message>& messages,
125 bool force,
126 const std::string& conversation_id) {
127 int current = counter.count_messages(messages);
128 int threshold = static_cast<int>(
129 static_cast<float>(counter.max_tokens) * config.threshold_percent);
130
131 if (!force && current < threshold) {
132 return {false, current, current};
133 }
134
135 if (!config.enabled) {
136 logger->warn("Context at {}/{} tokens, compaction disabled",
137 current, counter.max_tokens);
138 return {false, current, current};
139 }
140
141 logger->info("Compacting conversation ({} tokens)", current);
142
143 // Save full history before compacting (v1.8.8)
144 if (config.save_full_history && !conversation_id.empty()) {
145 save_snapshot(conversation_id, messages);
146 }
147
148 std::string summary;
149 int stripped = 0;
150 auto compacted = compact(messages, summary, stripped);
151
152 counter.clear_cache();
153 int new_count = counter.count_messages(compacted);
154
155 if (new_count >= current) {
156 logger->error("Compaction did not reduce tokens: {} -> {}",
157 current, new_count);
158 }
159
160 logger->info("Compacted {} -> {} tokens", current, new_count);
161 messages = std::move(compacted);
162
163 CompactionResult result;
164 result.compacted = true;
165 result.old_token_count = current;
166 result.new_token_count = new_count;
167 result.summary = summary;
168 result.preserved_messages = static_cast<int>(messages.size()) - 1;
169 result.messages_summarized = stripped;
170 return result;
171}

182std::vector<Message> CompactionManager::compact(
183 const std::vector<Message>& messages,
184 std::string& summary,
185 int& stripped_count) {
186 Message const* system_msg = nullptr;
187 size_t start = 0;
188 if (!messages.empty() && messages[0].role == "system") {
189 system_msg = &messages[0];
190 start = 1;
191 }
192
193 std::vector<const Message*> user_msgs;
194 std::vector<const Message*> assistant_msgs;
195 stripped_count = 0;
196
197 for (size_t i = start; i < messages.size(); ++i) {
198 const auto& msg = messages[i];
199 auto src = msg.metadata.find("source");
200 if (src != msg.metadata.end() && src->second == "user") {
201 user_msgs.push_back(&msg);
202 } else if (msg.role == "assistant") {
203 assistant_msgs.push_back(&msg);
204 } else {
205 ++stripped_count;
206 }
207 }
208
209 std::vector<Message> working(messages.begin() + static_cast<long>(start),
210 messages.end());
211 logger->info("Compact: {} user, {} assistant, {} stripped",
212 user_msgs.size(), assistant_msgs.size(), stripped_count);
213 summary = structured_summary(working);
214 Message summary_msg;
215 summary_msg.role = "user";
216 summary_msg.content = format_summary(
217 summary, static_cast<int>(working.size()));
218
219 std::vector<Message> result;
220 if (system_msg != nullptr) {
221 result.push_back(*system_msg);
222 }
223 result.push_back(std::move(summary_msg));
224 for (const auto* m : user_msgs) {
225 result.push_back(*m);
226 }
227 if (!assistant_msgs.empty()) {
228 result.push_back(*assistant_msgs.back());
229 }
230 return result;
231}

262std::string CompactionManager::extract_original_task(
263 const std::vector<Message>& messages) {
264 std::string task;
265 for (const auto& msg : messages) {
266 auto src = msg.metadata.find("source");
267 if (src != msg.metadata.end() && src->second == "user") {
268 task = msg.content;
269 break;
270 }
271 }
272 if (task.empty()) {
273 for (const auto& msg : messages) {
274 if (msg.role == "user" && !msg.content.empty()
275 && msg.content[0] != '[') {
276 task = msg.content;
277 break;
278 }
279 }
280 }
281 if (task.empty()) {
282 return "(no user message found)";
283 }
284 if (task.size() > 500) {
285 return task.substr(0, 500) + "...";
286 }
287 return task;
288}

297std::vector<std::pair<std::string, std::string>>
298CompactionManager::extract_tool_log(
299 const std::vector<Message>& messages) {
300 std::vector<std::pair<std::string, std::string>> log;
301 for (const auto& msg : messages) {
302 auto it = msg.metadata.find("tool_name");
303 if (it == msg.metadata.end()) {
304 continue;
305 }
306 const auto& name = it->second;
307 if (msg.content.rfind("[Previous:", 0) == 0) {
308 log.emplace_back(name, "(pruned)");
309 continue;
310 }
311 auto nl = msg.content.find('\n');
312 std::string brief = msg.content.substr(0, std::min(nl, size_t{100}));
313 log.emplace_back(name, brief);
314 }
315 return log;
316}

398static std::string json_escape(const std::string& input) {
399 std::ostringstream oss;
400 for (char c : input) {
401 if (c == '"') oss << "\\\"";
402 else if (c == '\\') oss << "\\\\";
