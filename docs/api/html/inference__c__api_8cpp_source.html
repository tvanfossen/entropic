150std::vector<entropic::Message> parse_messages_json(const char* json_str) {
151 std::vector<entropic::Message> messages;
152 auto arr = nlohmann::json::parse(json_str);
153 for (const auto& m : arr) {
154 entropic::Message msg;
155 msg.role = m.value("role", "user");
156 if (m.contains("content") && m["content"].is_array()) {
157 for (const auto& part : m["content"]) {
158 msg.content_parts.push_back(parse_content_part(part));
159 }
160 msg.content = entropic::extract_text(msg.content_parts);
161 } else {
162 msg.content = m.value("content", "");
163 }
164 messages.push_back(std::move(msg));
165 }
166 return messages;
167}

255ENTROPIC_EXPORT entropic_error_t entropic_inference_generate_streaming(
256 entropic_inference_backend_t backend,
257 const char* messages_json,
258 const char* params_json,
259 void (*on_token)(const char* token, size_t len, void* user_data),
260 void* user_data,
261 int* cancel_flag)
262{
263 try {
264 auto msgs = parse_messages_json(messages_json);
265 auto params = parse_params_json(params_json);
266 std::atomic<bool> cancel{false};
267
268 auto callback = [on_token, user_data, cancel_flag, &cancel]
269 (std::string_view token) {
270 on_token(token.data(), token.size(), user_data);
271 if (cancel_flag && *cancel_flag) {
272 cancel.store(true, std::memory_order_release);
273 }
274 };
275
276 auto result = to_backend(backend)->generate_streaming(
277 msgs, params, callback, cancel);
278 return result.ok() ? ENTROPIC_OK : result.error_code;
279 } catch (const std::exception& e) {
280 logger->error("inference_generate_streaming exception: {}", e.what());
281 return ENTROPIC_ERROR_GENERATE_FAILED;
282 }
