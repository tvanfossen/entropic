179std::string validate_routing(
180 const RoutingConfig& routing,
181 const ModelsConfig& models)
182{
183 std::string err;
184
185 if (models.tiers.empty()) {
186 /* no tiers — nothing to validate */
187 } else if (routing.enabled && !models.router.has_value()) {
188 err = "routing.enabled is true but models.router is not configured";
189 } else {
190 err = validate_fallback_tier(routing.fallback_tier, models.tiers);
191
192 if (err.empty()) {
193 err = validate_tier_map(routing.tier_map, models.tiers);
194 }
195 if (err.empty()) {
196 err = validate_handoff_rules(routing.handoff_rules, models.tiers);
197 }
198 }
199
200 return err;
201}

252static std::string validate_model_tiers(const ModelsConfig& models)
253{
254 std::string err = validate(models);
255
256 if (err.empty()) {
257 for (const auto& [name, tier] : models.tiers) {
258 err = validate(static_cast<const ModelConfig&>(tier));
259 if (!err.empty()) {
260 err = "models." + name + ": " + err;
261 break;
262 }
263 }
264 }
265
266 if (err.empty() && models.router.has_value()) {
267 err = validate(*models.router);
268 if (!err.empty()) {
269 err = "models.router: " + err;
270 }
271 }
272
273 return err;
274}

284std::string validate_config(
285 const ParsedConfig& config,
286 std::vector<std::string>& warnings)
287{
288 s_log->info("Config validation start");
289 std::string err = validate_model_tiers(config.models);
290
291 if (err.empty()) {
292 err = validate_routing(config.routing, config.models);
293 }
294 if (err.empty()) {
295 err = validate(config.compaction);
296 }
297 if (err.empty()) {
298 err = validate(config.prompt_cache);
299 }
300 if (err.empty()) {
301 auto w = warn_auto_chain_without_targets(
302 config.models.tiers, config.routing.handoff_rules);
303 if (!w.empty()) {
304 warnings.push_back(w);
305 }
306 }
307
308 if (err.empty()) {
309 s_log->info("Config validation passed ({} warning(s))",
310 warnings.size());
311 } else {
312 s_log->error("Config validation failed: {}", err);
313 }
314 return err;
315}
