38bool ExternalMCPClient::connect() {
39 if (!transport_->open()) {

43
44 if (!send_initialize()) {

92std::string ExternalMCPClient::execute(
93 const std::string& tool_name,
94 const std::string& args_json) {
95
96 if (!transport_->is_connected()) {
97 return build_response(
98 "Server '" + name_ + "' is disconnected. "
99 "Tool '" + name_ + "." + tool_name + "' unavailable.",
100 true);
101 }
102
103 nlohmann::json params;
104 params["name"] = tool_name;
105 try {
106 params["arguments"] = nlohmann::json::parse(args_json);
107 } catch (...) {
108 params["arguments"] = nlohmann::json::object();
109 }
110
111 auto request = build_request("tools/call", params.dump());
112 auto response = transport_->send_request(
113 request, DEFAULT_TIMEOUT_MS);
114
115 if (response.empty()) {
116 return build_response(
117 "Tool '" + name_ + "." + tool_name +
118 "' timed out or transport error.", true);
119 }
120
121 auto result_text = extract_tool_result(response);
122 return build_response(result_text);
123}

131std::pair<std::vector<std::string>, std::vector<std::string>>
132ExternalMCPClient::refresh_tools() {
133
134 std::set<std::string> old_names;
135 {
136 std::lock_guard<std::mutex> lock(tools_mutex_);
137 old_names.insert(cached_tool_names_.begin(),
138 cached_tool_names_.end());
139 }
140
141 query_tools();
142
143 std::set<std::string> new_names;
144 {
145 std::lock_guard<std::mutex> lock(tools_mutex_);
146 new_names.insert(cached_tool_names_.begin(),
147 cached_tool_names_.end());
148 }
149
150 std::vector<std::string> added;
151 std::vector<std::string> removed;
152 std::set_difference(new_names.begin(), new_names.end(),
153 old_names.begin(), old_names.end(),
154 std::back_inserter(added));
155 std::set_difference(old_names.begin(), old_names.end(),
156 new_names.begin(), new_names.end(),
157 std::back_inserter(removed));
158
159 logger->info("Server '{}' tools refreshed: +{} -{}",
160 name_, added.size(), removed.size());
161 return {added, removed};
162}

250bool ExternalMCPClient::query_tools() {
251 auto request = build_request("tools/list");
252 auto response = transport_->send_request(
253 request, INIT_TIMEOUT_MS);
254
255 if (response.empty()) {
256 return false;
257 }
258
259 try {
260 auto j = nlohmann::json::parse(response);
261 auto tools = j.at("result").at("tools");
262
263 // Prefix tool names with server name
264 std::vector<std::string> names;
265 for (auto& tool : tools) {
266 std::string orig = tool["name"].get<std::string>();
267 tool["name"] = name_ + "." + orig;
268 names.push_back(tool["name"].get<std::string>());
269 }
270
271 std::lock_guard<std::mutex> lock(tools_mutex_);
272 cached_tools_json_ = tools.dump();
273 cached_tool_names_ = std::move(names);
274 return true;
275 } catch (const nlohmann::json::exception& e) {
276 logger->error("Failed to parse tools/list from '{}': {}",
277 name_, e.what());
278 return false;
279 }
280}

289std::string ExternalMCPClient::extract_tool_result(
290 const std::string& response_json) {
291
292 try {
293 auto j = nlohmann::json::parse(response_json);
294 if (j.contains("error")) {
295 return "Error: " + j["error"]["message"]
296 .get<std::string>();
297 }
298
299 auto& content = j.at("result").at("content");
300 std::string text;
301 for (const auto& item : content) {
302 if (item.value("type", "") == "text") {
303 text += item.at("text").get<std::string>();
304 }
305 }
306 return text;
307 } catch (const nlohmann::json::exception& e) {
308 return "Error parsing response: " + std::string(e.what());
309 }
310}

320std::string ExternalMCPClient::build_response(
321 const std::string& result_text,
322 bool is_error) {
323
324 nlohmann::json resp;
325 resp["result"] = result_text;
326 // SECURITY: External servers CANNOT inject directives.
327 // Directives array is always empty for external tool results.
328 resp["directives"] = nlohmann::json::array();
329 if (is_error) {
330 resp["is_error"] = true;
331 }
