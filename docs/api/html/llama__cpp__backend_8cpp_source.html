206 prompt_cache_ = std::make_unique<PromptCache>(

360LogprobResult LlamaCppBackend::do_evaluate_logprobs(
361 const int32_t* tokens,
362 int n_tokens)
363{
364 int n_vocab = llama_vocab_n_tokens(vocab_);
365 LogprobResult result;
366 result.tokens.assign(tokens, tokens + n_tokens);
367 result.n_tokens = n_tokens;
368 result.n_logprobs = n_tokens - 1;
369 result.logprobs.reserve(result.n_logprobs);
370
371 auto* mem = llama_get_memory(ctx_);
372 llama_memory_clear(mem, true);
373
374 for (int i = 0; i < n_tokens; i++) {
375 llama_token tok = tokens[i];
376 llama_batch batch = llama_batch_get_one(&tok, 1);
377 int rc = llama_decode(ctx_, batch);
378 if (rc != 0) {
379 llama_memory_clear(mem, true);
380 throw std::runtime_error("llama_decode failed at logprob pos");
381 }
382 if (i < n_tokens - 1) {
383 const float* logits = llama_get_logits_ith(ctx_, -1);
384 float lp = extract_token_logprob(
385 logits, tokens[i + 1], n_vocab);
386 result.logprobs.push_back(lp);
387 }
388 }
389
390 float sum = 0.0f;
391 for (float lp : result.logprobs) { sum += lp; }
392 result.total_logprob = sum;
393 result.perplexity = std::exp(
394 -sum / static_cast<float>(result.n_logprobs));
395
396 llama_memory_clear(mem, true);
397 return result;
398}

440float LlamaCppBackend::extract_token_logprob(
441 const float* logits,
442 int32_t next_token,
443 int n_vocab)
444{
445 float max_logit = logits[0];
446 for (int v = 1; v < n_vocab; v++) {
447 if (logits[v] > max_logit) {
448 max_logit = logits[v];
449 }
450 }
451 float sum_exp = 0.0f;
452 for (int v = 0; v < n_vocab; v++) {
453 sum_exp += std::exp(logits[v] - max_logit);
454 }
455 float log_sum_exp = max_logit + std::log(sum_exp);
456 return logits[next_token] - log_sum_exp;
457}

474std::string LlamaCppBackend::apply_chat_template(
475 const std::vector<Message>& messages,
476 const GenerationParams& params) const
477{
478 // Convert to llama_chat_message array
479 std::vector<llama_chat_message> chat_msgs;
480 chat_msgs.reserve(messages.size());
481 for (const auto& msg : messages) {
482 chat_msgs.push_back({msg.role.c_str(), msg.content.c_str()});
483 }
484
485 // First call: measure required buffer size
486 int n = llama_chat_apply_template(
487 nullptr, chat_msgs.data(), chat_msgs.size(),
488 true, nullptr, 0);
489 if (n < 0) {
490 logger->error("llama_chat_apply_template failed (size query)");
491 // Fallback: concatenate messages directly
492 std::string fallback;
493 for (const auto& msg : messages) {
494 fallback += msg.role + ": " + msg.content + "\n";
495 }
496 return fallback;
497 }
498
499 std::vector<char> buf(static_cast<size_t>(n + 1));
500 int written = llama_chat_apply_template(
501 nullptr, chat_msgs.data(), chat_msgs.size(),
502 true, buf.data(), static_cast<int32_t>(buf.size()));
503 if (written < 0) {
504 logger->error("llama_chat_apply_template failed (render)");
505 return "";
506 }
507
508 return std::string(buf.data(), static_cast<size_t>(written));
509}

524llama_sampler* LlamaCppBackend::create_sampler(
525 const GenerationParams& params) const
526{
527 llama_sampler_chain_params chain_params = llama_sampler_chain_default_params();
528 llama_sampler* chain = llama_sampler_chain_init(chain_params);
529
530 // Grammar constraint (applied first)
531 if (!params.grammar.empty()) {
532 llama_sampler* grammar = llama_sampler_init_grammar(
533 vocab_, params.grammar.c_str(), "root");
534 if (grammar) {
535 llama_sampler_chain_add(chain, grammar);
536 }
537 }
538
539 // Repetition penalty
540 if (params.repeat_penalty != 1.0f) {
541 llama_sampler_chain_add(chain,
542 llama_sampler_init_penalties(
543 64, params.repeat_penalty, 0.0f, 0.0f));
544 }
545
546 // Temperature
547 if (params.temperature > 0.0f) {
548 llama_sampler_chain_add(chain,
549 llama_sampler_init_temp(params.temperature));
550 }
551
552 // Top-K
553 if (params.top_k > 0) {
554 llama_sampler_chain_add(chain,
555 llama_sampler_init_top_k(params.top_k));
556 }
557
558 // Top-P (nucleus sampling)
559 if (params.top_p < 1.0f) {
560 llama_sampler_chain_add(chain,
561 llama_sampler_init_top_p(params.top_p, 1));
562 }
563
564 // Final distribution sampler
565 llama_sampler_chain_add(chain, llama_sampler_init_dist(0));
566
567 return chain;
568}

579bool LlamaCppBackend::run_prefill(const std::vector<llama_token>& tokens) {
580 llama_memory_clear(llama_get_memory(ctx_), true);
581
582 const int n_batch = config().n_batch;
583 const int n_tokens = static_cast<int>(tokens.size());
584
585 for (int i = 0; i < n_tokens; i += n_batch) {
586 int chunk = std::min(n_batch, n_tokens - i);
587 std::vector<llama_token> slice(
588 tokens.begin() + i, tokens.begin() + i + chunk);
589 llama_batch batch = llama_batch_get_one(
590 slice.data(), static_cast<int32_t>(chunk));
591 if (llama_decode(ctx_, batch) != 0) {
592 logger->error("Prefill decode failed at offset {}", i);
593 return false;
594 }
595 }
596 return true;
597}

609std::string LlamaCppBackend::step_token(
610 llama_sampler* sampler,
611 std::string& generated,
612 std::function<void(std::string_view)>& on_token,
613 const std::vector<std::string>& stop)
614{
615 llama_token new_token = llama_sampler_sample(sampler, ctx_, -1);
616
617 if (new_token == llama_vocab_eos(vocab_)
618 || llama_vocab_is_eog(vocab_, new_token)) {
619 return "eos";
620 }
621
622 std::string piece = detokenize(new_token);
623 generated += piece;
624 if (on_token) {
625 on_token(std::string_view(piece));
626 }
627 if (check_stop_sequences(generated, stop)) {
628 return "stop";
629 }
630
631 llama_token tok = new_token;
632 llama_batch single = llama_batch_get_one(&tok, 1);
633 return (llama_decode(ctx_, single) == 0) ? "continue" : "error";
634}

646GenerationResult LlamaCppBackend::decode_loop(
647 const std::vector<llama_token>& tokens,
648 const GenerationParams& params,
649 std::function<void(std::string_view)> on_token,
650 std::atomic<bool>* cancel)
651{
652 GenerationResult result;
653 llama_sampler* sampler = create_sampler(params);
654
655 if (!run_prefill(tokens)) {
656 result.error_code = ENTROPIC_ERROR_GENERATE_FAILED;
657 result.error_message = "Prefill decode failed";
658 result.finish_reason = "error";
659 llama_sampler_free(sampler);
660 return result;
661 }
662
663 std::string generated;
664 int n_generated = 0;
665
666 while (n_generated < params.max_tokens) {
667 bool cancelled = cancel && cancel->load(std::memory_order_acquire);
668 if (cancelled) {
669 result.finish_reason = "cancelled";
670 result.error_code = ENTROPIC_ERROR_CANCELLED;
671 break;
672 }
673
674 auto status = step_token(sampler, generated, on_token, params.stop);
675 if (status == "continue") {
676 ++n_generated;
677 } else {
678 result.finish_reason = (status == "error") ? "error" : "stop";
679 if (status == "error") {
680 result.error_code = ENTROPIC_ERROR_GENERATE_FAILED;
681 }
682 break;
683 }
684 }
685
686 if (n_generated >= params.max_tokens && result.finish_reason.empty()) {
687 result.finish_reason = "length";
688 }
689
690 llama_sampler_free(sampler);
691 result.content = generated;
692 result.token_count = n_generated;
693 return result;
694}

724bool LlamaCppBackend::restore_cached_prefix(
725 const CacheEntry* cached,
726 const std::vector<llama_token>& tokens)
727{
728 llama_memory_clear(llama_get_memory(ctx_), true);
729
730 size_t restored = llama_state_seq_set_data(
731 ctx_, cached->data.data(), cached->data_size, 0);
732 if (restored == 0) {
733 logger->warn("KV state restore failed, falling back to full prefill");
734 return false;
735 }
736
737 int prefix_len = cached->token_count;
738 if (prefix_len >= static_cast<int>(tokens.size())) {
739 return true;
740 }
741
742 std::vector<llama_token> remaining(
743 tokens.begin() + prefix_len, tokens.end());
744 llama_batch batch = llama_batch_get_one(
745 remaining.data(), static_cast<int32_t>(remaining.size()));
746
747 bool ok = (llama_decode(ctx_, batch) == 0);
748 if (!ok) {
749 logger->error("Decode of remaining tokens after cache restore failed");
750 }
751 return ok;
752}

819bool LlamaCppBackend::run_prefill_cached(
820 const std::vector<llama_token>& tokens,
821 const std::string& system_prompt,
822 const std::vector<Message>& messages,
823 const GenerationParams& params)
824{
825 bool cache_enabled = prompt_cache_
826 && prompt_cache_config_.enabled
827 && !system_prompt.empty();
828
829 // Try cache restore if enabled
830 bool restored = false;
831 CacheKey key{0};
832
833 if (cache_enabled) {
834 key = PromptCache::make_key(
835 system_prompt, config().path.string());
836 const CacheEntry* cached = prompt_cache_->lookup(key);
837
838 if (cached) {
839 if (prompt_cache_config_.log_hits) {
840 logger->info("Prompt cache HIT: {} bytes, {} prefix tokens",
841 cached->data_size, cached->token_count);
842 }
843 restored = restore_cached_prefix(cached, tokens);
844 } else if (prompt_cache_config_.log_hits) {
845 logger->info("Prompt cache MISS: processing full prompt");
846 }
847 }
848
849 // Fall back to full prefill if not restored from cache
850 if (!restored && !run_prefill(tokens)) {
851 return false;
852 }
853
854 // Save prefix to cache on miss
855 if (cache_enabled && !restored) {
856 int prefix_tokens = compute_prefix_token_count(messages, params);
857 if (prefix_tokens > 0) {
858 save_prefix_to_cache(key, prefix_tokens);
859 }
860 }
861 return true;
862}

874GenerationResult LlamaCppBackend::do_generate(
875 const std::vector<Message>& messages,
876 const GenerationParams& params)
877{
878 auto t0 = entropic::log::now();
879 std::string prompt = apply_chat_template(messages, params);
880 auto tokens = tokenize(prompt, true);
881 std::string sys = extract_system_prompt(messages);
882
883 logger->info("Generate: {} input tokens, max_tokens={}",
884 tokens.size(), params.max_tokens);
885 log_sampler_config(params);
886
887 GenerationResult result;
888 llama_sampler* sampler = create_sampler(params);
889
890 if (!run_prefill_cached(tokens, sys, messages, params)) {
891 llama_sampler_free(sampler);
892 return prefill_error();
893 }
894
895 std::string generated;
896 int n_generated = 0;
897 std::function<void(std::string_view)> no_cb = nullptr;
898
899 while (n_generated < params.max_tokens) {
900 auto status = step_token(
901 sampler, generated, no_cb, params.stop);
902 if (status == "continue") { ++n_generated; }
903 else {
904 result.finish_reason =
905 (status == "error") ? "error" : "stop";
906 if (status == "error") {
907 result.error_code = ENTROPIC_ERROR_GENERATE_FAILED;
908 }
909 break;
910 }
911 }
912
913 if (n_generated >= params.max_tokens
914 && result.finish_reason.empty()) {
915 result.finish_reason = "length";
916 }
917
918 llama_sampler_free(sampler);
919 result.content = generated;
920 result.token_count = n_generated;
921 finalize_result(result, t0);
922 return result;
923}

935GenerationResult LlamaCppBackend::do_generate_streaming(
936 const std::vector<Message>& messages,
937 const GenerationParams& params,
938 std::function<void(std::string_view token)> on_token,
939 std::atomic<bool>& cancel)
940{
941 auto t0 = entropic::log::now();
942 auto prompt = apply_chat_template(messages, params);
943 auto tokens = tokenize(prompt, true);
944 auto sys = extract_system_prompt(messages);
945 logger->info("Stream: {} input tokens, max_tokens={}",
946 tokens.size(), params.max_tokens);
947 log_sampler_config(params);
948
949 GenerationResult result;
950 auto* sampler = create_sampler(params);
951 if (!run_prefill_cached(tokens, sys, messages, params)) {
952 llama_sampler_free(sampler);
953 return prefill_error();
954 }
955 std::string generated;
956 int n_generated = 0;
957 while (n_generated < params.max_tokens) {
958 if (cancel.load(std::memory_order_acquire)) {
959 result.finish_reason = "cancelled";
960 result.error_code = ENTROPIC_ERROR_CANCELLED;
961 break;
962 }
963 auto status = step_token(
964 sampler, generated, on_token, params.stop);
965 if (status == "continue") { ++n_generated; }
966 else {
967 result.finish_reason =
968 (status == "error") ? "error" : "stop";
969 if (status == "error") {
970 result.error_code = ENTROPIC_ERROR_GENERATE_FAILED;
971 }
972 break;
973 }
974 }
975 if (n_generated >= params.max_tokens
976 && result.finish_reason.empty()) {
977 result.finish_reason = "length";
978 }
979 llama_sampler_free(sampler);
980 result.content = generated;
981 result.token_count = n_generated;
982 finalize_result(result, t0);
983 return result;
984}

1030bool LlamaCppBackend::do_supports(BackendCapability cap) const {
1031 int idx = static_cast<int>(cap);
1032 int count = static_cast<int>(BackendCapability::_COUNT);
1033 if (idx < 0 || idx >= count) {
1034 return false;
1035 }
1036
1037 // Static capabilities: true = always supported
1038 static constexpr bool always[] = {
1039 false, false, true, true, true, true,
1040 false, true, true, false, false, true,
1041 };
1042
1043 // Dynamic capabilities override the static table
1044 bool result = always[idx];
1045 if (!result) {
1046 result = (cap == BackendCapability::KV_CACHE && !is_recurrent())
1047 || (cap == BackendCapability::HIDDEN_STATE && is_recurrent())
1048 || (cap == BackendCapability::VISION
1049 && !config().mmproj_path.empty())
1050 || (cap == BackendCapability::SPECULATIVE_DECODING
1051 && !is_recurrent());
1052 }
1053 return result;
1054}
