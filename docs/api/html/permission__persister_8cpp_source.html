95static bool list_contains(const std::vector<std::string>& lines,
96 size_t start, std::string_view pattern) {
97 std::string needle = "- " + std::string(pattern);
98 for (size_t i = start + 1; i < lines.size(); ++i) {
99 auto pos = lines[i].find_first_not_of(' ');
100 if (pos == std::string::npos) continue;
101 auto trimmed = lines[i].substr(pos);
102 if (!trimmed.starts_with("- ")) break;
103 if (trimmed == needle) return true;
104 }
105 return false;
106}

176bool PermissionPersister::save_permission(std::string_view pattern,
177 bool allow) {
178 std::lock_guard lock(write_mutex_);
179
180 auto lines = split_lines(read_file(config_path_));
181 const char* list_key = allow ? "allow" : "deny";
182 std::string perms_key = "permissions:";
183 std::string list_line = std::string(" ") + list_key + ":";
184 std::string item = " - " + std::string(pattern);
185
186 auto perms_it = std::find(lines.begin(), lines.end(), perms_key);
187 if (perms_it == lines.end()) {
188 lines.push_back(perms_key);
189 lines.push_back(list_line);
190 lines.push_back(item);
191 } else {
192 auto list_it = std::find(perms_it, lines.end(), list_line);
193 if (list_it == lines.end()) {
194 auto pos = find_section_end(lines, perms_it);
195 lines.insert(pos, item);
196 lines.insert(pos, list_line);
197 } else {
198 size_t idx = static_cast<size_t>(
199 std::distance(lines.begin(), list_it));
200 if (list_contains(lines, idx, pattern)) {
201 return true;
202 }
203 lines.insert(find_list_end(lines, idx), item);
204 }
205 }
206
207 if (!write_file(config_path_, join_lines(lines))) {
208 logger->error("Failed to write permission to {}",
209 config_path_.string());
210 return false;
211 }
212
213 logger->info("Saved permission {}: {}",
214 allow ? "allow" : "deny", pattern);
215 return true;
216}
