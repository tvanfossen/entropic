78static std::pair<std::string, int> run_popen(
79 const std::string& full_cmd) {
80
81 FILE* pipe = popen(full_cmd.c_str(), "r"); // NOLINT
82 if (pipe == nullptr) {
83 return {"Failed to open process", -1};
84 }
85
86 std::string output;
87 std::array<char, 4096> buf{};
88 while (fgets(buf.data(), buf.size(), pipe) != nullptr) {
89 output += buf.data();
90 }
91
92 int status = pclose(pipe); // NOLINT
93 int exit_code = WEXITSTATUS(status);
94 return {output, exit_code};
95}
