40std::vector<std::string> interleave_examples(
41 const std::vector<TierDescriptor>& tiers)
42{
43 struct Queue {
44 int digit;
45 const std::vector<std::string>* examples;
46 };
47
48 std::vector<Queue> queues;
49 for (size_t i = 0; i < tiers.size(); ++i) {
50 if (!tiers[i].examples.empty()) {
51 queues.push_back(
52 {static_cast<int>(i + 1), &tiers[i].examples});
53 }
54 }
55
56 std::vector<std::string> lines;
57 size_t round_idx = 0;
58 while (true) {
59 bool added = false;
60 for (const auto& q : queues) {
61 if (round_idx < q.examples->size()) {
62 lines.push_back("\"" + (*q.examples)[round_idx]
63 + "\" -> "
64 + std::to_string(q.digit));
65 added = true;
66 }
67 }
68 if (!added) {
69 break;
70 }
71 ++round_idx;
72 }
73 return lines;
74}

124std::string build_classification_prompt(
125 const std::vector<TierDescriptor>& tiers,
126 const std::string& message,
127 const std::vector<std::string>& history,
128 const std::vector<std::string>& recent_tiers)
129{
130 std::string result = "Classify the message. Reply with the number only.\n\n";

132
133 if (!recent_tiers.empty()) {
134 result += "Recent tiers: " + join(recent_tiers, " -> ") + "\n\n";
135 }
136
137 if (!history.empty()) {
138 size_t start = history.size() > 5 ? history.size() - 5 : 0;
139 std::vector<std::string> recent(history.begin() + static_cast<long>(start), history.end());
140 result += "Recent messages: " + join(recent, " | ") + "\n\n";
141 }
142
143 for (const auto& line : interleave_examples(tiers)) {
144 result += line + "\n";
145 }
146
147 result += "\"" + message + "\" -> ";
148 logger->info("Classification prompt: {} tiers, {} history, "
149 "{} chars",
150 tiers.size(), history.size(), result.size());
151 return result;
152}
