53bool InferenceBackend::load(const ModelConfig& config) {
54 std::lock_guard<std::mutex> lock(transition_mutex_);
55
56 if (state() != ModelState::COLD) {
57 logger->info("[VRAM] load() no-op: already {}", state_name(state()));
58 return true;
59 }
60
61 // Hook: ON_MODEL_LOAD — can cancel (v1.9.1)
62 bool cancelled = fire_model_load_hook(config);
63 if (cancelled) {
64 return false;
65 }
66
67 logger->info("[VRAM] Loading: {}", config.path.string());
68 auto start = entropic::log::now();
69
70 config_ = config;
71 bool ok = do_load(config);
72 if (!ok) {
73 logger->error("[VRAM] Load failed: {}", last_error_);
74 } else {
75 state_.store(ModelState::WARM, std::memory_order_release);
76 logger->info("[VRAM] Warm in {:.2f}ms", entropic::log::elapsed_ms(start, entropic::log::now()));
77 }
78 return ok;
79}

87bool InferenceBackend::activate() {
88 std::lock_guard<std::mutex> lock(transition_mutex_);
89
90 if (state() == ModelState::ACTIVE) {
91 logger->info("[VRAM] activate() no-op: already ACTIVE");
92 return true;
93 }
94 if (state() != ModelState::WARM) {
95 logger->error("[VRAM] activate() failed: not WARM ({})", state_name(state()));
96 return false;
97 }
98
99 logger->info("[VRAM] Activating");
100 auto start = entropic::log::now();
101 bool ok = do_activate();
102 if (!ok) {
103 logger->error("[VRAM] Activate failed: {}", last_error_);
104 } else {
105 state_.store(ModelState::ACTIVE, std::memory_order_release);
106 logger->info("[VRAM] Active in {:.2f}ms", entropic::log::elapsed_ms(start, entropic::log::now()));
107 }
108 return ok;
109}

138void InferenceBackend::unload() {

140
141 // Hook: ON_MODEL_UNLOAD — informational (v1.9.1)
142 if (hooks_.fire_info != nullptr) {

148
149 logger->info("[VRAM] Unloading from {}", state_name(state()));
150
151 do_unload();

274LogprobResult InferenceBackend::evaluate_logprobs(
275 const int32_t* tokens,
276 int n_tokens)
277{
278 if (!is_active()) {
279 logger->error("evaluate_logprobs: model not ACTIVE (state={})",
280 state_name(state()));
281 throw std::runtime_error("Model must be ACTIVE for evaluation");
282 }
283
284 if (n_tokens < 2) {
285 logger->error("evaluate_logprobs: need >= 2 tokens, got {}",
286 n_tokens);
287 throw std::runtime_error(
288 "Need at least 2 tokens for logprob evaluation");
289 }
290
291 std::lock_guard<std::mutex> lock(eval_mutex_);
292
293 logger->info("evaluate_logprobs: {} tokens, first=[{},{},{}...]",
294 n_tokens, tokens[0],
295 n_tokens > 1 ? tokens[1] : 0,
296 n_tokens > 2 ? tokens[2] : 0);
297 auto start = entropic::log::now();
298
299 LogprobResult result = do_evaluate_logprobs(tokens, n_tokens);
300
301 result.total_logprob = 0.0f;
302 for (float lp : result.logprobs) {
303 result.total_logprob += lp;
304 }
305 float mean_lp = result.total_logprob /
306 static_cast<float>(result.n_logprobs);
307 result.perplexity = std::exp(-mean_lp);
308
309 auto ms = entropic::log::elapsed_ms(start, entropic::log::now());
310 logger->info("evaluate_logprobs: perplexity={:.2f}, "
311 "total_lp={:.4f}, {:.2f}ms",
312 result.perplexity, result.total_logprob, ms);
313 for (int i = 0; i < result.n_logprobs; ++i) {
314 logger->info(" logprob[{}]={:.4f}", i, result.logprobs[i]);
315 }
316
317 return result;
318}

348bool InferenceBackend::fire_model_load_hook(const ModelConfig& config) {
349 if (hooks_.fire_pre == nullptr) {
350 return false;
351 }
352 std::string json = "{\"model_path\":\""
353 + config.path.string() + "\"}";
354 char* mod = nullptr;
355 int rc = hooks_.fire_pre(hooks_.registry,
356 ENTROPIC_HOOK_ON_MODEL_LOAD, json.c_str(), &mod);
357 free(mod);
358 if (rc != 0) {
359 logger->info("[VRAM] ON_MODEL_LOAD hook cancelled");
360 }
361 return rc != 0;
362}
