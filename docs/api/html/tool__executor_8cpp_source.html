57std::vector<Message> ToolExecutor::process_tool_calls(
58 LoopContext& ctx,
59 const std::vector<ToolCall>& tool_calls) {
60 logger->info("Processing {} tool calls", tool_calls.size());
61 ctx.state = AgentState::WAITING_TOOL;
62 fire_state_callback(ctx);
63
64 auto limited = sort_tool_calls(tool_calls);
65 truncate_to_limit(limited);
66
67 std::vector<Message> results;
68 for (const auto& call : limited) {
69 auto msgs = process_single_call(ctx, call);
70 for (auto& m : msgs) {
71 results.push_back(std::move(m));
72 }
73 if (should_stop_batch(ctx, results)) {
74 break;
75 }
76 }
77
78 ctx.consecutive_errors = 0;
79 return results;
80}

89std::vector<ToolCall> ToolExecutor::sort_tool_calls(
90 const std::vector<ToolCall>& calls) {
91 auto sorted = calls;
92 std::stable_sort(sorted.begin(), sorted.end(),
93 [](const ToolCall& a, const ToolCall& b) {
94 bool a_delegate = (a.name == "entropic.delegate");
95 bool b_delegate = (b.name == "entropic.delegate");
96 return !a_delegate && b_delegate;
97 });
98 return sorted;

153bool ToolExecutor::check_approval(const ToolCall& call) {
154 auto args_json = serialize_args(call);
155 bool auto_ok = loop_config_.auto_approve_tools
156 || server_manager_.is_explicitly_allowed(
157 call.name, args_json);
158
159 bool approved = auto_ok;
160 if (!approved && callbacks_.on_tool_call != nullptr) {
161 auto call_json = serialize_tool_call(call);
162 callbacks_.on_tool_call(call_json.c_str(),
163 callbacks_.user_data);
164 approved = true;
165 }
166
167 // Hook: ON_PERMISSION_CHECK — informational (v1.9.1)
168 if (hook_iface_.fire_info != nullptr) {
169 std::string perm = approved ? "allowed" : "denied";
170 std::string json = "{\"tool_name\":\""
171 + call.name + "\",\"permission\":\"" + perm + "\"}";
172 hook_iface_.fire_info(hook_iface_.registry,
173 ENTROPIC_HOOK_ON_PERMISSION_CHECK, json.c_str());
174 }
175
176 if (!approved) {
177 logger->warn("No approval callback — denying: {}", call.name);
178 }
179 return approved;
180}

209std::pair<Message, std::string> ToolExecutor::execute_tool(
210 LoopContext& ctx, const ToolCall& call) {
211
212 auto args_json = serialize_args(call);
213 logger->info("[TOOL CALL] {}", call.name);
214
215 if (callbacks_.on_tool_start != nullptr) {
216 auto call_json = serialize_tool_call(call);
217 callbacks_.on_tool_start(call_json.c_str(),
218 callbacks_.user_data);
219 }
220
221 auto start = std::chrono::steady_clock::now();
222 auto result_json = server_manager_.execute(
223 call.name, args_json);
224 auto end = std::chrono::steady_clock::now();
225 auto ms = std::chrono::duration_cast<
226 std::chrono::milliseconds>(end - start).count();
227
228 ctx.metrics.tool_calls++;
229
230 // Parse result
231 std::string result_text;
232 try {
233 auto j = nlohmann::json::parse(result_json);
234 result_text = j.value("result", result_json);
235 } catch (...) {
236 result_text = result_json;
237 }
238
239 logger->info("[TOOL COMPLETE] {} -> {} chars ({}ms)",
240 call.name, result_text.size(), ms);
241
242 fire_tool_complete_callback(call, result_text, ms);
243
244 Message msg;
245 msg.role = "user";
246 msg.content = result_text;
247 msg.metadata["tool_call_id"] = call.id;
248 msg.metadata["tool_name"] = call.name;
249
250 return {std::move(msg), result_json};
251}

376std::optional<Message> ToolExecutor::check_mcp_authorization(
377 const LoopContext& ctx,
378 const ToolCall& call) const {
379 if (auth_mgr_ == nullptr) {
380 return std::nullopt;
381 }
382 auto identity = ctx.locked_tier.empty()
383 ? "lead" : ctx.locked_tier;
384 auto required = server_manager_.get_required_access_level(
385 call.name);
386 if (!auth_mgr_->is_enforced(identity) ||
387 auth_mgr_->check_access(identity, call.name, required)) {
388 return std::nullopt;
389 }
390 auto level_str = mcp_access_level_name(required);
391 logger->warn("MCP key denied: {} requires {} for {}",
392 call.name, level_str, identity);
393 Message msg;
394 msg.role = "user";
395 msg.content =
396 "Tool `" + call.name + "` was denied: identity `"
397 + identity + "` lacks " + level_str
398 + " access.\n\n"
399 "Your MCP key set does not authorize this tool. "
400 "Use `entropic.delegate` to hand off to an identity "
401 "that has the required access.";
402 return msg;
403}

455std::vector<Message> ToolExecutor::process_single_call(
456 LoopContext& ctx, const ToolCall& call) {
457
458 auto rejection = check_call_preconditions(ctx, call);
459 if (rejection.has_value()) {
460 logger->info("Tool '{}' rejected by precondition", call.name);
461 return {std::move(*rejection)};
462 }
463
464 // Hook: PRE_TOOL_CALL — can cancel (v1.9.1)
465 if (hook_iface_.fire_pre != nullptr) {
466 std::string json = "{\"tool_name\":\""
467 + call.name + "\"}";
468 char* mod = nullptr;
469 int rc = hook_iface_.fire_pre(hook_iface_.registry,
470 ENTROPIC_HOOK_PRE_TOOL_CALL, json.c_str(), &mod);
471 free(mod);
472 if (rc != 0) {
473 return {create_denied_message(call, "Cancelled by hook")};
474 }
475 }
476
477 auto exec_start = std::chrono::steady_clock::now();
478 auto [msg, raw_result] = execute_tool(ctx, call);
479 auto exec_ms = std::chrono::duration<double, std::milli>(
480 std::chrono::steady_clock::now() - exec_start).count();
481 ctx.effective_tool_calls++;
482 msg.metadata["added_at_iteration"] =
483 std::to_string(ctx.metrics.iterations);
484 record_tool_call(ctx, call, raw_result);
485
486 // Hook: POST_TOOL_CALL — enriched context (v1.9.5)
487 if (hook_iface_.fire_post != nullptr) {
488 auto json = build_post_tool_json(call, raw_result, exec_ms);
489 char* out = nullptr;
490 hook_iface_.fire_post(hook_iface_.registry,
491 ENTROPIC_HOOK_POST_TOOL_CALL, json.c_str(), &out);
492 free(out);
493 }
494
495 logger->info("Tool '{}' executed: {:.0f}ms, result={} chars",
496 call.name, exec_ms, raw_result.size());
497 run_post_tool_hooks(ctx);
498
499 return {std::move(msg)};
500}

643std::string ToolExecutor::build_post_tool_json(
644 const ToolCall& call,
645 const std::string& raw_result,
646 double elapsed_ms) {
647 nlohmann::json ctx;
648 ctx["tool_name"] = call.name;
649 ctx["args"] = nlohmann::json::parse(serialize_args(call));
650 ctx["elapsed_ms"] = elapsed_ms;
651 try {
652 auto sr = nlohmann::json::parse(raw_result);
653 ctx["result"] = sr.value("result", raw_result);
654 ctx["directives"] = sr.value(
655 "directives", nlohmann::json::array());
656 } catch (...) {
657 ctx["result"] = raw_result;
658 ctx["directives"] = nlohmann::json::array();
659 }
660 return ctx.dump();
