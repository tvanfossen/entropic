116 int fds[6] = {-1, -1, -1, -1, -1, -1};

221bool StdioTransport::spawn_child(
222 int stdin_r, int stdout_w, int stderr_w,
223 const std::vector<std::string>& env_strs) {
224
225 posix_spawn_file_actions_t actions;
226 posix_spawn_file_actions_init(&actions);
227 posix_spawn_file_actions_adddup2(&actions, stdin_r, STDIN_FILENO);
228 posix_spawn_file_actions_adddup2(&actions, stdout_w, STDOUT_FILENO);
229 posix_spawn_file_actions_adddup2(&actions, stderr_w, STDERR_FILENO);
230
231 // Build argv: [command, args..., nullptr]
232 std::vector<char*> argv;
233 argv.push_back(const_cast<char*>(command_.c_str()));
234 for (const auto& arg : args_) {
235 argv.push_back(const_cast<char*>(arg.c_str()));
236 }
237 argv.push_back(nullptr);
238
239 // Build envp
240 std::vector<char*> envp;
241 for (const auto& e : env_strs) {
242 envp.push_back(const_cast<char*>(e.c_str()));
243 }
244 envp.push_back(nullptr);
245
246 int err = posix_spawnp(
247 &child_pid_, command_.c_str(), &actions,
248 nullptr, argv.data(), envp.data());
249
250 posix_spawn_file_actions_destroy(&actions);
251
252 if (err != 0) {
253 logger->error("posix_spawnp failed for '{}': {}",
254 command_, strerror(err));
255 child_pid_ = -1;
256 return false;
257 }
258 return true;
259}

267std::vector<std::string> StdioTransport::build_env() const {
268 std::map<std::string, std::string> merged;
269
270 // Copy parent environment
271 for (char** ep = environ; ep && *ep; ++ep) {
272 std::string entry(*ep);
273 auto eq = entry.find('=');
274 if (eq != std::string::npos) {
275 merged[entry.substr(0, eq)] = entry.substr(eq + 1);
276 }
277 }
278
279 // Apply overrides
280 for (const auto& [key, val] : env_) {
281 merged[key] = val;
282 }
283
284 std::vector<std::string> result;
285 result.reserve(merged.size());
286 for (const auto& [key, val] : merged) {
287 result.push_back(key + "=" + val);
288 }
289 return result;
290}

324 auto remaining = std::chrono::duration_cast<

328 if (remaining.count() <= 0) {

344std::string StdioTransport::read_line(int fd, uint32_t timeout_ms) {
345 std::string line;
346 auto deadline = std::chrono::steady_clock::now() +
347 std::chrono::milliseconds(timeout_ms);
348
349 while (true) {
350 int ready = poll_until_ready(fd, deadline);
351 if (ready <= 0) {
352 if (ready == 0) {
353 logger->warn("Read timeout after {}ms", timeout_ms);
354 }
355 return "";
356 }
357
358 char ch = 0;
359 ssize_t n = ::read(fd, &ch, 1);
360 if (n <= 0) {
361 return "";
362 }
363 if (ch == '\n') {
364 return line;
365 }
366 line += ch;
367 }

399void StdioTransport::terminate_child() {
400 if (child_pid_ <= 0) {
401 return;
402 }
403
404 ::kill(child_pid_, SIGTERM);
405
406 // Wait up to 3 seconds for graceful exit
407 constexpr int max_wait_ms = 3000;
408 constexpr int poll_interval_ms = 50;
409 int waited = 0;
410 while (waited < max_wait_ms) {
411 int status = 0;
412 pid_t result = ::waitpid(child_pid_, &status, WNOHANG);
413 if (result == child_pid_) {
414 child_pid_ = -1;
415 return;
416 }
417 std::this_thread::sleep_for(
418 std::chrono::milliseconds(poll_interval_ms));
419 waited += poll_interval_ms;
420 }
421
422 // Force kill
423 ::kill(child_pid_, SIGKILL);
424 ::waitpid(child_pid_, nullptr, 0);
425 logger->warn("Force-killed child PID {}", child_pid_);
426 child_pid_ = -1;
427}
