40entropic_error_t HookRegistry::register_hook(
41 entropic_hook_point_t point,
42 entropic_hook_callback_t callback,
43 void* user_data,
44 int priority) {
45 if (!is_valid_point(point)) {
46 return ENTROPIC_ERROR_INVALID_CONFIG;
47 }
48
49 auto idx = static_cast<size_t>(point);
50 std::unique_lock lock(mutexes_[idx]);
51
52 HookEntry entry{callback, user_data, priority};
53 auto& vec = entries_[idx];
54
55 // Insert sorted by priority (ascending)
56 auto pos = std::lower_bound(
57 vec.begin(), vec.end(), entry,
58 [](const HookEntry& a, const HookEntry& b) {
59 return a.priority < b.priority;
60 });
61 vec.insert(pos, entry);
62
63 logger->info("Registered hook: point={} priority={} total={}",
64 static_cast<int>(point), priority, vec.size());
65 return ENTROPIC_OK;
66}

77entropic_error_t HookRegistry::deregister_hook(
78 entropic_hook_point_t point,
79 entropic_hook_callback_t callback,
80 void* user_data) {
81 if (!is_valid_point(point)) {
82 return ENTROPIC_ERROR_INVALID_CONFIG;
83 }
84
85 auto idx = static_cast<size_t>(point);
86 std::unique_lock lock(mutexes_[idx]);
87
88 auto& vec = entries_[idx];
89 auto it = std::find_if(vec.begin(), vec.end(),
90 [callback, user_data](const HookEntry& e) {
91 return e.callback == callback
92 && e.user_data == user_data;
93 });
94
95 if (it != vec.end()) {
96 vec.erase(it);
97 logger->info("Deregistered hook: point={}", static_cast<int>(point));
98 }
99 return ENTROPIC_OK;

125int HookRegistry::fire_pre(
126 entropic_hook_point_t point,
127 const char* context_json,
128 char** out_json) {
129 *out_json = nullptr;
130
131 if (!is_valid_point(point)) {
132 return 0;
133 }
134
135 auto entries = snapshot(point);
136 logger->info("fire_pre: point={}, {} handler(s)",
137 static_cast<int>(point), entries.size());
138 char* accumulated = nullptr;
139 int result = 0;
140
141 for (const auto& entry : entries) {
142 const char* input = accumulated ? accumulated : context_json;
143 char* modified = nullptr;
144 int rc = 0;
145
146 try {
147 rc = entry.callback(point, input, &modified, entry.user_data);
148 } catch (const std::exception& e) {
149 logger->warn("Pre-hook threw for point={}: {}",
150 static_cast<int>(point), e.what());
151 continue;
152 } catch (...) {
153 logger->warn("Pre-hook threw unknown exception for point={}",
154 static_cast<int>(point));
155 continue;
156 }
157
158 if (rc != 0) {
159 free(accumulated);
160 free(modified);
161 accumulated = nullptr;
162 result = rc;
163 break;
164 }
165
166 if (modified != nullptr) {
167 free(accumulated);
168 accumulated = modified;
169 }
170 }
171
172 *out_json = accumulated;
173 return result;
174}
