147std::string AuditLogger::utc_timestamp() {
148 auto now = std::chrono::system_clock::now();
149 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
150 now.time_since_epoch()) % 1000;
151 auto time = std::chrono::system_clock::to_time_t(now);
152 std::tm utc{};
153 gmtime_r(&time, &utc);
154 std::ostringstream ss;
155 ss << std::put_time(&utc, "%Y-%m-%dT%H:%M:%S")
156 << '.' << std::setfill('0') << std::setw(3) << ms.count() << 'Z';
157 return ss.str();
158}

180void AuditLogger::rotate_files() {
181 file_.close();
182 auto base = config_.log_dir / "audit.jsonl";
183 // Shift existing rotated files (N → N+1), drop oldest
184 for (size_t i = config_.max_files; i >= 1; --i) {
185 auto src = base.string() + "." + std::to_string(i);
186 auto dst = base.string() + "." + std::to_string(i + 1);
187 if (i == config_.max_files) {
188 std::filesystem::remove(src);
189 } else if (std::filesystem::exists(src)) {
190 std::filesystem::rename(src, dst);
191 }
192 }
193 std::filesystem::rename(base, base.string() + ".1");
194 file_.open(base, std::ios::app);
195 current_file_size_ = 0;
196 logger->info("Audit log rotated");
197}

209int AuditLogger::hook_callback(
210 entropic_hook_point_t /*hook_point*/,
211 const char* context_json,
212 char** modified_json,
213 void* user_data) {
214 *modified_json = nullptr;
215 auto* ctx = static_cast<AuditHookContext*>(user_data);
216 if (ctx == nullptr || ctx->logger == nullptr) {
217 return 0;
218 }
219 try {
220 auto j = nlohmann::json::parse(context_json);
221 AuditEntry entry;
222 entry.tool_name = j.value("tool_name", "");
223 entry.params_json = j.contains("args")
224 ? j["args"].dump() : "{}";
225 entry.result_content = j.value("result", "");
226 entry.elapsed_ms = j.value("elapsed_ms", 0.0);
227 entry.directives_json = j.contains("directives")
228 ? j["directives"].dump() : "[]";
229 populate_from_hook_context(entry, *ctx);
230 ctx->logger->record(entry);
231 } catch (const std::exception& e) {
232 logger->error("Audit hook callback failed: {}", e.what());
233 }
234 return 0;
235}
