127bool MCPKeySet::deserialize(const std::string& json) {
128 nlohmann::json arr;
129 try {
130 arr = nlohmann::json::parse(json);
131 } catch (const nlohmann::json::exception& e) {
132 logger->warn("Deserialize failed: {}", e.what());
133 return false;
134 }
135 if (!arr.is_array()) {
136 logger->warn("Deserialize: expected JSON array");
137 return false;
138 }
139 std::lock_guard<std::mutex> lock(key_mutex_);
140 keys_.clear();
141 for (const auto& entry : arr) {
142 if (!entry.contains("pattern") || !entry.contains("level")) {
143 logger->warn("Skipping entry: missing fields");
144 continue;
145 }
146 MCPAccessLevel level{};
147 auto level_str = entry["level"].get<std::string>();
148 if (!parse_mcp_access_level(level_str, level)) {
149 logger->warn("Skipping entry: unknown level '{}'",
150 level_str);
151 continue;
152 }
153 keys_[entry["pattern"].get<std::string>()] = level;
154 }
155 return true;
156}

183MCPAccessLevel MCPKeySet::find_best_match(
184 const std::string& tool_name) const {
185 auto result = MCPAccessLevel::NONE;
186 // 1. Exact match (most specific)
187 auto it = keys_.find(tool_name);
188 if (it != keys_.end()) {
189 result = it->second;
190 } else if (auto wc = server_wildcard(tool_name);
191 !wc.empty() &&
192 (it = keys_.find(wc)) != keys_.end()) {
193 // 2. Server wildcard (e.g., "filesystem.*")
194 result = it->second;
195 } else if ((it = keys_.find("*")) != keys_.end()) {
196 // 3. Full wildcard
197 result = it->second;
198 }
199 return result;
200}
