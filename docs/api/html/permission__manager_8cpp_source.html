100bool PermissionManager::pattern_matches(
101 const std::string& tool_name,
102 const std::string& full_pattern,
103 const std::string& permission_pattern) {
104
105 // Split permission pattern at ':'
106 auto colon = permission_pattern.find(':');
107 std::string pattern_tool = (colon != std::string::npos)
108 ? permission_pattern.substr(0, colon)
109 : permission_pattern;
110
111 // Tool name must match the tool portion
112 if (fnmatch(pattern_tool.c_str(), tool_name.c_str(), 0) != 0) {
113 return false;
114 }
115
116 // If no arg pattern, tool match is sufficient
117 if (colon == std::string::npos) {
118 return true;
119 }
120
121 // Full pattern must match
122 return fnmatch(permission_pattern.c_str(),
123 full_pattern.c_str(), 0) == 0;
124}
