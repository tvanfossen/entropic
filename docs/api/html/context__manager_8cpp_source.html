59std::pair<int, int> ContextManager::prune_tool_results(
60 LoopContext& ctx,
61 int keep_recent) {
62 std::vector<size_t> indices;
63 for (size_t i = 0; i < ctx.messages.size(); ++i) {
64 auto it = ctx.messages[i].metadata.find("tool_name");
65 if (it != ctx.messages[i].metadata.end()) {
66 indices.push_back(i);
67 }
68 }
69
70 size_t cut = 0;
71 if (keep_recent > 0
72 && indices.size() > static_cast<size_t>(keep_recent)) {
73 cut = indices.size() - static_cast<size_t>(keep_recent);
74 } else {
75 return {0, 0};
76 }
77
78 int pruned = 0;
79 int freed = 0;
80 for (size_t j = 0; j < cut; ++j) {
81 auto& msg = ctx.messages[indices[j]];
82 if (msg.content.rfind("[Previous:", 0) == 0) {
83 continue;
84 }
85 auto name_it = msg.metadata.find("tool_name");
86 std::string name = (name_it != msg.metadata.end())
87 ? name_it->second : "unknown";
88 int chars = static_cast<int>(msg.content.size());
89 freed += chars;
90 msg.content = "[Previous: " + name + " result — "
91 + std::to_string(chars) + " chars, pruned]";
92 ++pruned;
93 }
94
95 if (pruned > 0) {
96 compaction_.counter.clear_cache();
97 logger->info("Pruned {} tool result(s), freed {} chars",
98 pruned, freed);
99 }
100 return {pruned, freed};
101}

109void ContextManager::prune_old_tool_results(LoopContext& ctx) {
110 int ttl = compaction_.config.tool_result_ttl;
111 int current = ctx.metrics.iterations;
112 int pruned = 0;
113
114 for (auto& msg : ctx.messages) {
115 auto tn = msg.metadata.find("tool_name");
116 if (tn == msg.metadata.end()) {
117 continue;
118 }
119 if (msg.content.rfind("[Previous:", 0) == 0) {
120 continue;
121 }
122 auto ai = msg.metadata.find("added_at_iteration");
123 if (ai == msg.metadata.end()) {
124 continue;
125 }
126 int added = 0;
127 try { added = std::stoi(ai->second); }
128 catch (...) { continue; }
129 if (current - added < ttl) {
130 continue;
131 }
132 int chars = static_cast<int>(msg.content.size());
133 msg.content = "[Previous: " + tn->second + " result — "
134 + std::to_string(chars) + " chars, pruned]";
135 ++pruned;
136 }
137
138 if (pruned > 0) {
139 compaction_.counter.clear_cache();
140 logger->info("[AUTO-PRUNE] Pruned {} results (TTL={})", pruned, ttl);
141 }
142}

150void ContextManager::inject_context_warning(LoopContext& ctx) {
151 float threshold = compaction_.config.warning_threshold_percent;
152 float usage = compaction_.counter.usage_percent(ctx.messages);
153 if (usage < threshold) {
154 return;
155 }
156
157 auto last = ctx.metadata.find("last_warning_iteration");
158 std::string iter_str = std::to_string(ctx.metrics.iterations);
159 if (last != ctx.metadata.end() && last->second == iter_str) {
160 return;
161 }
162
163 int max_tok = compaction_.counter.max_tokens;
164 int cur_tok = compaction_.counter.count_messages(ctx.messages);
165 int pct = static_cast<int>(usage * 100.0f);
166
167 Message warning;
168 warning.role = "user";
169 warning.content = "[CONTEXT WARNING] Context at "
170 + std::to_string(pct) + "% capacity ("
171 + std::to_string(cur_tok) + "/" + std::to_string(max_tok)
172 + " tokens). Capture findings with entropic.todo if needed,"
173 " then call entropic.prune_context.";
174 ctx.messages.push_back(std::move(warning));
175 ctx.metadata["last_warning_iteration"] = iter_str;
176 logger->info("[WARNING] Context at {}% — warning injected", pct);
177}

186void ContextManager::check_compaction(
187 LoopContext& ctx,
188 bool force) {
189 int cur = compaction_.counter.count_messages(ctx.messages);
190 int max = compaction_.counter.max_tokens;
191 if (max > 0) {
192 int pct = (cur * 100) / max;
193 logger->info("Context: {}/{} tokens ({}%)", cur, max, pct);
194 }
195
196 // Hook: ON_PRE_COMPACT — can cancel compaction (v1.9.1)
197 if (hook_iface_.fire_pre != nullptr) {
198 int tok = compaction_.counter.count_messages(ctx.messages);
199 std::string json = "{\"token_count\":"
200 + std::to_string(tok) + ",\"force\":"
201 + (force ? "true" : "false") + "}";
202 char* mod = nullptr;
203 int rc = hook_iface_.fire_pre(hook_iface_.registry,
204 ENTROPIC_HOOK_ON_PRE_COMPACT, json.c_str(), &mod);
205 free(mod);
206 if (rc != 0) {
207 logger->info("ON_PRE_COMPACT hook cancelled compaction");
208 return;
209 }
210 }
211
212 auto result = compaction_.check_and_compact(
213 ctx.messages, force, ctx.conversation_id);
214
215 if (result.compacted) {
216 logger->info("Compacted: {} -> {} tokens",
217 result.old_token_count, result.new_token_count);
218 if (callbacks_.on_compaction != nullptr) {
219 std::string json = "{\"old\":"
220 + std::to_string(result.old_token_count)
221 + ",\"new\":" + std::to_string(result.new_token_count)
222 + "}";
223 callbacks_.on_compaction(json.c_str(), callbacks_.user_data);
224 }
225
226 // Hook: ON_POST_COMPACT (v1.9.1)
227 if (hook_iface_.fire_post != nullptr) {
228 std::string json = "{\"tokens_before\":"
229 + std::to_string(result.old_token_count)
230 + ",\"tokens_after\":"
231 + std::to_string(result.new_token_count) + "}";
232 char* out = nullptr;
233 hook_iface_.fire_post(hook_iface_.registry,
234 ENTROPIC_HOOK_ON_POST_COMPACT, json.c_str(), &out);
235 free(out);
236 }
237
238 if (hooks_.after_compaction) {
239 hooks_.after_compaction(ctx);
240 }
241 }
