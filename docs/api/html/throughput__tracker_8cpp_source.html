26void ThroughputTracker::record(int tokens_generated, int64_t elapsed_ms) {
27 if (tokens_generated < kMinTokens || elapsed_ms <= 0) {
28 return;
29 }
30
31 double sample_tok_s =
32 static_cast<double>(tokens_generated) / (static_cast<double>(elapsed_ms) / 1000.0);
33
34 std::lock_guard<std::mutex> lock(mutex_);
35 if (samples_.load(std::memory_order_relaxed) == 0) {
36 ewma_tok_s_.store(sample_tok_s, std::memory_order_relaxed);
37 } else {
38 double prev = ewma_tok_s_.load(std::memory_order_relaxed);
39 ewma_tok_s_.store(
40 kAlpha * sample_tok_s + (1.0 - kAlpha) * prev,
41 std::memory_order_relaxed);
42 }
43 samples_.fetch_add(1, std::memory_order_relaxed);
44 logger->info("Throughput sample: {:.1f} tok/s, EWMA={:.1f} tok/s, "
45 "{} samples",
46 sample_tok_s,
47 ewma_tok_s_.load(std::memory_order_relaxed),
48 samples_.load(std::memory_order_relaxed));
49}
