22std::string BundledModels::load(const std::filesystem::path& path)
23{
24 std::string err;
25
26 auto content = read_file(path);
27 if (content.empty()) {
28 err = "cannot read " + path.string();
29 }
30
31 ryml::Tree tree;
32 ryml::ConstNodeRef root;
33 if (err.empty()) {
34 tree = ryml::parse_in_arena(
35 ryml::to_csubstr(path.string()),
36 ryml::to_csubstr(content));
37 root = tree.rootref();
38
39 if (!root.is_map()) {
40 err = "bundled_models.yaml root is not a mapping";
41 }
42 }
43
44 if (err.empty()) {
45 for (auto child : root) {
46 BundledModelEntry entry;
47 entry.key = to_string(child.key());
48 extract(child, "name", entry.name);
49 extract(child, "url", entry.url);
50 extract(child, "size_gb", entry.size_gb);
51 extract(child, "adapter", entry.adapter);
52 extract(child, "description", entry.description);
53
54 if (entry.name.empty()) {
55 err = "bundled model '" + entry.key + "' missing 'name'";
56 break;
57 }
58
59 s_log->info("Registered bundled model: {} -> {} ({:.1f} GB)",
60 entry.key, entry.name, entry.size_gb);
61 entries_[entry.key] = std::move(entry);
62 }
63 }
64
65 return err;
66}
