150static bool json_read_string(JsonCursor& c, std::string& out) {
151 json_skip_ws(c);
152 if (c.pos >= c.data.size() || c.data[c.pos] != '"') {
153 return false;
154 }
155 ++c.pos;
156 out.clear();
157 while (c.pos < c.data.size() && c.data[c.pos] != '"') {
158 if (c.data[c.pos] == '\\' && c.pos + 1 < c.data.size()) {
159 out += json_unescape_char(c.data[c.pos + 1]);
160 c.pos += 2;
161 } else {
162 out += c.data[c.pos++];
163 }
164 }
165 if (c.pos >= c.data.size()) return false;
166 ++c.pos; // skip closing '"'
167 return true;
168}

176static void json_skip_composite(JsonCursor& c) {
177 char open = c.data[c.pos];
178 char close = (open == '[') ? ']' : '}';
179 int depth = 1;
180 ++c.pos;
181 while (c.pos < c.data.size() && depth > 0) {
182 char ch = c.data[c.pos];
183 if (ch == '"') {
184 std::string dummy;
185 json_read_string(c, dummy);
186 continue;
187 }
188 if (ch == open) ++depth;
189 else if (ch == close) --depth;
190 ++c.pos;
191 }
192}

200static void json_skip_value(JsonCursor& c) {
201 json_skip_ws(c);
202 if (c.pos >= c.data.size()) return;
203 char ch = c.data[c.pos];
204 if (ch == '"') {
205 std::string dummy;
206 json_read_string(c, dummy);
207 return;
208 }
209 if (ch == '[' || ch == '{') {
210 json_skip_composite(c);
211 return;
212 }
213 while (c.pos < c.data.size()
214 && c.data[c.pos] != ',' && c.data[c.pos] != '}'
215 && c.data[c.pos] != ']') {
216 ++c.pos;
217 }
218}

229static bool json_read_content_part_field(JsonCursor& c,

252static bool json_parse_content_part(JsonCursor& c,
253 std::string& type_val,
254 std::string& text_val) {
255 while (c.pos < c.data.size() && c.data[c.pos] != '}') {

258 if (!json_read_content_part_field(c, type_val, text_val)) {
259 return false;

341static bool json_extract_text_from_array(JsonCursor& c,
342 std::string& text) {
343 if (c.pos >= c.data.size() || c.data[c.pos] != '[') {
344 return false;
345 }
346 ++c.pos; // skip '['
347 text.clear();
348
349 bool ok = true;
350 char ch = json_skip_to_element(c);
351 while (ok && ch == '{') {
352 ok = json_read_one_content_part(c, text);
353 ch = ok ? json_skip_to_element(c) : '\0';
354 }
355 if (ok && ch == ']') { ++c.pos; }
356 return ok && ch == ']';

371static bool json_read_field(JsonCursor& c, Message& msg) {
372 std::string key;
373 bool ok = json_read_string(c, key);
374 json_skip_ws(c);
375 if (ok && c.pos < c.data.size() && c.data[c.pos] == ':') {
376 ++c.pos;
377 } else {
378 ok = false;
379 }
380
381 if (!ok) return false;
382
383 if (key == "role") {
384 ok = json_read_string(c, msg.role);
385 } else if (key == "content") {
386 json_skip_ws(c);
387 if (c.pos < c.data.size() && c.data[c.pos] == '[') {
388 ok = json_extract_text_from_array(c, msg.content);
389 } else {
390 ok = json_read_string(c, msg.content);
391 }
392 } else {
393 json_skip_value(c);
394 }
395 return ok;
396}

406static bool json_parse_message(JsonCursor& c, Message& msg) {
407 while (c.pos < c.data.size()) {
408 json_skip_ws(c);
409 if (c.data[c.pos] == '}') { ++c.pos; return true; }
410 if (c.data[c.pos] == ',') { ++c.pos; continue; }
411 if (!json_read_field(c, msg)) return false;
412 }
413 return false;

424static bool json_parse_array(JsonCursor& c,
425 std::vector<Message>& messages) {
426 bool ok = true;
427 while (ok && c.pos < c.data.size()) {
428 json_skip_ws(c);
429 if (c.data[c.pos] == ']') break;
430 if (c.data[c.pos] == ',') { ++c.pos; continue; }
431 ok = (c.data[c.pos] == '{');
432 if (!ok) break;
433 ++c.pos;
434
435 Message msg;
436 ok = json_parse_message(c, msg);
437 if (ok) messages.push_back(std::move(msg));
438 }
439 return ok;
440}

642CompactorFn CompactorRegistry::wrap_c_compactor(
643 entropic_compactor_fn compactor,
644 void* user_data) {
645 return [compactor, user_data](
646 const std::vector<Message>& messages,
647 const CompactionConfig& config,
648 const std::string& identity) -> CompactionResult {
649 auto msg_json = serialize_messages(messages);
650 auto cfg_json = serialize_config(config, identity, 0);
651
652 char* out_messages = nullptr;
653 char* out_summary = nullptr;
654 int rc = compactor(
655 msg_json.c_str(), cfg_json.c_str(),
656 &out_messages, &out_summary, user_data);
657
658 CompactionResult result;
659 if (rc != 0) {
660 result.compacted = false;
661 free(out_messages);
662 free(out_summary);
663 return result;
664 }
665
666 result.compacted = true;
667 if (out_summary != nullptr) {
668 result.summary = out_summary;
669 free(out_summary);
670 }
671 if (out_messages != nullptr) {
672 parse_messages(out_messages, result.messages);
673 result.preserved_messages =
674 static_cast<int>(result.messages.size());
675 free(out_messages);
676 }
677 return result;
678 };
