51bool SessionLogger::initialize() {
52 std::filesystem::create_directories(config_.log_dir);
53
54 auto session_path = config_.log_dir / "session.log";
55 auto model_path = config_.log_dir / "session_model.log";
56
57 try {
58 session_logger_ = spdlog::rotating_logger_mt(
59 "entropic_session",
60 session_path.string(),
61 config_.max_file_size,
62 config_.max_files);
63 session_logger_->set_pattern(
64 "[%Y-%m-%d %H:%M:%S] [%l] %v");
65
66 model_logger_ = spdlog::rotating_logger_mt(
67 "entropic_model",
68 model_path.string(),
69 config_.max_file_size,
70 config_.max_files);
71 model_logger_->set_pattern("[%Y-%m-%d %H:%M:%S] %v");
72
73 session_logger_->info("Session log started");
74 model_logger_->info("Model output log started");
75 return true;
76 } catch (const spdlog::spdlog_ex& ex) {
77 logger->error("SessionLogger init failed: {}", ex.what());
78 return false;
79 }
