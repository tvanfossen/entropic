52
53 if (!handle) { return ENTROPIC_ERROR_INVALID_HANDLE; }
54 if (!identity_json) { return ENTROPIC_ERROR_INVALID_ARGUMENT; }
55 if (!handle->identity_manager) { return ENTROPIC_ERROR_INVALID_STATE; }
56
57 try {
58 auto names = handle->identity_manager->list();
59 if (names.empty()) { return ENTROPIC_ERROR_IDENTITY_NOT_FOUND; }
60 auto* cfg = handle->identity_manager->get(names[0]);
61 if (!cfg) { return ENTROPIC_ERROR_IDENTITY_NOT_FOUND; }
62 nlohmann::json j;
63 j["name"] = cfg->name;
64 j["origin"] = (cfg->origin == entropic::IdentityOrigin::STATIC)
65 ? "static" : "dynamic";
66 *identity_json = strdup(j.dump().c_str());
67 return ENTROPIC_OK;
68 } catch (const std::exception& e) {
69 handle->last_error = e.what();
70 return ENTROPIC_ERROR_INTERNAL;
