66bool ProfileRegistry::register_profile(const GPUResourceProfile& profile) {

104GPUResourceProfile ProfileRegistry::get(const std::string& name) const {
105 std::lock_guard<std::mutex> lock(mutex_);
106 auto it = profiles_.find(name);
107 if (it != profiles_.end()) {
108 return it->second;
109 }
110
111 logger->warn("Profile '{}' not found, falling back to 'balanced'", name);
112 auto fallback = profiles_.find("balanced");
113 if (fallback != profiles_.end()) {

118 return GPUResourceProfile{};
