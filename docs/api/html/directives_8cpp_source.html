38DirectiveResult DirectiveProcessor::process(
39 LoopContext& ctx,
40 const std::vector<const Directive*>& directives) {
41 logger->info("Processing {} directive(s)", directives.size());
42 DirectiveResult result;
43 int processed = 0;
44 for (const auto* directive : directives) {
45 if (directive == nullptr) {
46 continue;
47 }
48
49 auto it = handlers_.find(static_cast<int>(directive->type));
50 bool has_handler = it != handlers_.end();
51
52 // Hook: ON_DIRECTIVE or ON_CUSTOM_DIRECTIVE (v1.9.1)
53 if (!fire_directive_hook(directive, has_handler)) {
54 continue; // suppressed by hook
55 }
56
57 if (has_handler) {
58 logger->info("Directive type={}, handler invoked",
59 static_cast<int>(directive->type));
60 it->second(ctx, *directive, result);
61 ++processed;
62 if (result.stop_processing) {
63 break;
64 }
65 }
66 }
67 logger->info("Directives: {}/{} processed", processed,
68 directives.size());
69 return result;
70}
