278ValidationResult ConstitutionalValidator::apply_revisions(
279 ValidationResult result,
280 const CritiqueResult& initial_critique,
281 const char* messages_json) {
282 if (config_.max_revisions <= 0) {
283 return result;
284 }
285
286 auto critique = initial_critique;
287 for (int i = 0; i < config_.max_revisions; ++i) {
288 auto revised = attempt_revision(
289 result.content, critique, messages_json);
290 result.content = revised;
291 result.was_revised = true;
292 result.revision_count = i + 1;
293
294 critique = run_critique(revised);
295 result.final_critique = critique;
296
297 if (critique.compliant) {
298 return result;
299 }
300 }
301
302 logger->warn("Constitutional validation: max revisions ({}) "
303 "exhausted, returning last output",
304 config_.max_revisions);
305 return result;
306}

334CritiqueResult ConstitutionalValidator::run_critique(
335 const std::string& content) {
336 if (inference_ == nullptr || inference_->generate == nullptr) {
337 logger->warn("Constitutional validation: no inference "
338 "interface, skipping critique");
339 return {};
340 }
341
342 auto messages = build_critique_messages(content);
343 auto params = build_critique_params();
344 char* result_json = nullptr;
345
346 int rc = inference_->generate(
347 messages.c_str(), params.c_str(),
348 &result_json, inference_->backend_data);
349
350 if (rc != 0 || result_json == nullptr) {
351 logger->warn("Constitutional validation: critique generation "
352 "failed (rc={})", rc);
353 return {};
354 }
355
356 std::string raw(result_json);
357 if (inference_->free_fn != nullptr) {
358 inference_->free_fn(result_json);
359 }
360
361 return parse_critique(raw);
362}

440static std::string json_escape(const std::string& s) {
441 std::string out;
442 out.reserve(s.size() + 16);
443 for (char c : s) {
444 switch (c) {
445 case '"': out += "\\\""; break;
446 case '\\': out += "\\\\"; break;
447 case '\n': out += "\\n"; break;
448 case '\r': out += "\\r"; break;
449 case '\t': out += "\\t"; break;
450 default: out += c; break;
451 }
452 }
453 return out;
454}

517std::string ConstitutionalValidator::inject_feedback_into_messages(
518 const std::string& original,
519 const std::string& feedback,
520 const char* messages_json) const {
521 std::string base = (messages_json != nullptr)
522 ? std::string(messages_json)
523 : "[]";
524
525 // Strip trailing ]
526 if (!base.empty() && base.back() == ']') {
527 base.pop_back();
528 }
529
530 // Add comma if there were existing messages
531 if (base.size() > 1) {
532 base += ",";
533 }
534
535 base += "{\"role\":\"assistant\",\"content\":\"";
536 base += json_escape(original);
537 base += "\"},{\"role\":\"user\",\"content\":\"";
538 base += json_escape(feedback);
539 base += "\"}]";
540 return base;

605std::string ConstitutionalValidator::extract_json_string(
606 const char* json, const char* key) {
607 if (json == nullptr || key == nullptr) {
608 return {};
609 }
610
611 std::string needle = std::string("\"") + key + "\"";
612 const char* pos = strstr(json, needle.c_str());
613 if (pos == nullptr) {
614 return {};
615 }
616
617 return extract_string_after_colon(pos + needle.size());
618}

627std::string ConstitutionalValidator::extract_string_after_colon(
628 const char* pos) {
629 // Skip whitespace and colon
630 while (*pos == ' ' || *pos == ':' || *pos == '\t') {
631 ++pos;
632 }
633 if (*pos != '"') {
634 return {};
635 }
636 ++pos; // skip opening quote
637
638 std::string value;
639 while (*pos != '\0' && *pos != '"') {
640 if (*pos == '\\' && *(pos + 1) != '\0') {
641 ++pos;
642 switch (*pos) {
643 case 'n': value += '\n'; break;
644 case 't': value += '\t'; break;
645 case 'r': value += '\r'; break;
646 default: value += *pos; break;
647 }
648 } else {
649 value += *pos;
650 }
651 ++pos;
652 }
653 return value;
654}

684void ConstitutionalValidator::extract_violations(
685 const std::string& json, CritiqueResult& result) {
686 size_t search_pos = 0;
687
688 while (true) {
689 auto v = extract_next_violation(json, search_pos);
690 if (!v.has_value()) {
691 break;
692 }
693 result.violations.push_back(std::move(v.value()));
694 }

705std::optional<Violation>
706ConstitutionalValidator::extract_next_violation(
707 const std::string& json, size_t& pos) {
708 auto rule_pos = json.find("\"rule\"", pos);
709 if (rule_pos == std::string::npos) {
710 return std::nullopt;
711 }
712
713 Violation v;
714 v.rule = extract_json_string(
715 json.c_str() + rule_pos, "rule");
716 v.excerpt = extract_json_string(
717 json.c_str() + rule_pos, "excerpt");
718 v.explanation = extract_json_string(
719 json.c_str() + rule_pos, "explanation");
720
721 pos = rule_pos + 1;
722 return v;
723}
