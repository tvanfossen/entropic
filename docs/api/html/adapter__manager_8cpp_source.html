72bool AdapterManager::load(
73 const std::string& name,
74 const std::filesystem::path& adapter_path,
75 llama_model* model,
76 float scale)
77{
78 std::lock_guard<std::mutex> lock(adapter_mutex_);
79
80 bool precondition_failed =
81 adapters_.find(name) != adapters_.end() || !model;
82 if (precondition_failed) {
83 logger->error("Cannot load adapter '{}': {}",
84 name, !model ? "base model is null" : "duplicate name");
85 return false;
86 }
87
88 auto t_start = now();
89 auto* lora = llama_adapter_lora_init(model, adapter_path.c_str());
90 if (!lora) {
91 logger->error("llama_adapter_lora_init failed for '{}' at {}",
92 name, adapter_path.string());
93 return false;
94 }
95
96 AdapterEntry entry;
97 entry.name = name;
98 entry.path = adapter_path;
99 entry.handle = lora;
100 entry.model = model;
101 entry.scale = scale;
102 entry.state = AdapterState::WARM;
103 adapters_.emplace(name, std::move(entry));
104
105 logger->info("Loaded adapter '{}' from {} in {:.1f}ms (scale={:.2f})",
106 name, adapter_path.string(),
107 elapsed_ms(t_start, now()), scale);
108 return true;
109}

124void AdapterManager::unload(const std::string& name, llama_context* ctx) {
125 std::lock_guard<std::mutex> lock(adapter_mutex_);
126
127 auto it = adapters_.find(name);
128 if (it == adapters_.end()) {
129 return;
130 }
131
132 auto& entry = it->second;
133 if (entry.state == AdapterState::HOT && ctx) {
134 clear_adapters(ctx);
135 active_name_.clear();
136 }
137
138 if (entry.handle) {
139 llama_adapter_lora_free(entry.handle);
140 entry.handle = nullptr;
141 }
142
143 entry.state = AdapterState::COLD;
144 logger->info("Unloaded adapter '{}'", name);
145
146 adapters_.erase(it);
147}

163bool AdapterManager::activate(const std::string& name, llama_context* ctx) {
164 std::lock_guard<std::mutex> lock(adapter_mutex_);
165
166 auto it = adapters_.find(name);
167 bool cannot_activate =
168 it == adapters_.end() || it->second.state == AdapterState::COLD;
169 if (it != adapters_.end() && it->second.state == AdapterState::HOT) {
170 return true; // Already active — no-op
171 }
172 if (cannot_activate) {
173 logger->error("Cannot activate adapter '{}': {}",
174 name, it == adapters_.end() ? "not found" : "state is COLD");
175 return false;
176 }
177
178 auto& entry = it->second;
179
180 // Mark previous HOT adapter as WARM
181 if (!active_name_.empty() && active_name_ != name) {
182 auto active_it = adapters_.find(active_name_);
183 if (active_it != adapters_.end()) {
184 active_it->second.state = AdapterState::WARM;
185 }
186 }
187
188 apply_adapter(ctx, entry.handle, entry.scale);
189 entry.state = AdapterState::HOT;
190 active_name_ = name;
191
192 logger->info("Activated adapter '{}' (scale={:.2f})", name, entry.scale);
193 return true;
194}

240bool AdapterManager::swap(const std::string& name, llama_context* ctx) {
241 std::lock_guard<std::mutex> lock(adapter_mutex_);
242
243 auto it = adapters_.find(name);
244 bool cannot_swap = it == adapters_.end()
245 || it->second.state == AdapterState::COLD;
246 if (active_name_ == name && !cannot_swap) {
247 return true; // Already active — no-op
248 }
249 if (cannot_swap || !fire_swap_hook(active_name_, name, it->second.path)) {
250 logger->error("Cannot swap to adapter '{}': {}",
251 name, cannot_swap ? "not found or COLD" : "cancelled by hook");
252 return false;
253 }
254
255 auto t_start = now();
256
257 // Mark current HOT as WARM
258 if (!active_name_.empty()) {
259 auto active_it = adapters_.find(active_name_);
260 if (active_it != adapters_.end()) {
261 active_it->second.state = AdapterState::WARM;
262 }
263 }
264
265 // Apply target
266 auto& target = it->second;
267 apply_adapter(ctx, target.handle, target.scale);
268 target.state = AdapterState::HOT;
269
270 std::string previous = active_name_;
271 active_name_ = name;
272
273 logger->info("Swapped adapter '{}' -> '{}' in {:.1f}ms",
274 previous, name, elapsed_ms(t_start, now()));
275 return true;
276}

291void AdapterManager::unload_all_for_model(
292 llama_model* model, llama_context* ctx)
293{
294 std::lock_guard<std::mutex> lock(adapter_mutex_);
295
296 bool cleared_context = false;
297 std::vector<std::string> to_remove;
298
299 for (auto& [name, entry] : adapters_) {
300 if (entry.model != model) {
301 continue;
302 }
303
304 if (entry.state == AdapterState::HOT
305 && ctx && !cleared_context)
306 {
307 clear_adapters(ctx);
308 cleared_context = true;
309 }
310
311 if (entry.handle) {
312 llama_adapter_lora_free(entry.handle);
313 entry.handle = nullptr;
314 }
315
316 entry.state = AdapterState::COLD;
317 to_remove.push_back(name);
318 }
319
320 for (const auto& name : to_remove) {
321 if (name == active_name_) {
322 active_name_.clear();
323 }
324 adapters_.erase(name);
325 }
326
327 logger->info("Unloaded {} adapter(s) for model", to_remove.size());
328}

435bool AdapterManager::fire_swap_hook(
436 const std::string& current,
437 const std::string& target,
438 const std::filesystem::path& target_path)
439{
440 if (!hooks_.fire_pre || !hooks_.registry) {
441 return true; // No hook registered — proceed
442 }
443
444 nlohmann::json ctx;
445 ctx["current_adapter"] = current;
446 ctx["target_adapter"] = target;
447 ctx["adapter_path"] = target_path.string();
448 std::string ctx_str = ctx.dump();
449
450 char* modified = nullptr;
451 int rc = hooks_.fire_pre(
452 hooks_.registry,
453 ENTROPIC_HOOK_ON_ADAPTER_SWAP,
454 ctx_str.c_str(),
455 &modified);
456
457 if (modified) {
458 free(modified);
459 }
460
461 return rc == 0;
462}
