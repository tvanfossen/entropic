182bool SqliteDatabase::initialize() {
183 std::lock_guard lock(mutex_);
184
185 // Ensure parent directory exists
186 auto parent = db_path_.parent_path();
187 if (!parent.empty()) {
188 std::filesystem::create_directories(parent);
189 }
190
191 int rc = sqlite3_open_v2(
192 db_path_.c_str(), &db_,
193 SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE | SQLITE_OPEN_FULLMUTEX,
194 nullptr);
195
196 if (rc != SQLITE_OK) {
197 logger->error("Failed to open database {}: {}",
198 db_path_.string(), sqlite3_errmsg(db_));
199 return false;
200 }
201
202 // Enable foreign keys and WAL mode
203 sqlite3_exec(db_, "PRAGMA foreign_keys = ON", nullptr, nullptr, nullptr);
204 sqlite3_exec(db_, "PRAGMA journal_mode = WAL", nullptr, nullptr, nullptr);
205
206 if (!run_migrations()) {
207 logger->error("Migration failed for {}", db_path_.string());
208 return false;
209 }
210
211 logger->info("Database initialized: {}", db_path_.string());
212 return true;
213}

268bool SqliteDatabase::execute(std::string_view sql,
269 std::function<void(sqlite3_stmt*)> binder) {
270 std::lock_guard lock(mutex_);
271 if (!db_) return false;
272
273 auto* stmt = prepare(sql);
274 if (!stmt) return false;
275
276 if (binder) binder(stmt);
277
278 int rc = sqlite3_step(stmt);
279 sqlite3_finalize(stmt);
280 bool ok = (rc == SQLITE_DONE);
281 if (!ok) {
282 logger->error("SQL execute failed: {}", sqlite3_errmsg(db_));
283 }

391static std::vector<std::string> get_applied_migrations(sqlite3* db) {
392 std::vector<std::string> applied;
393 sqlite3_stmt* stmt = nullptr;
394 int rc = sqlite3_prepare_v2(db,
395 "SELECT name FROM migrations", -1, &stmt, nullptr);
396 if (rc != SQLITE_OK) return applied;
397
398 while (sqlite3_step(stmt) == SQLITE_ROW) {
399 auto* text = sqlite3_column_text(stmt, 0);
400 if (text) {
401 applied.emplace_back(reinterpret_cast<const char*>(text));
402 }
403 }
404 sqlite3_finalize(stmt);
405 return applied;

416static bool apply_migration(sqlite3* db, const Migration& mig) {
417 logger->info("Running migration: {}", mig.name);
418
419 char* err_msg = nullptr;
420 int rc = sqlite3_exec(db, mig.sql, nullptr, nullptr, &err_msg);
421 if (rc != SQLITE_OK) {
422 logger->error("Migration {} failed: {}",
423 mig.name, err_msg ? err_msg : "unknown");
424 sqlite3_free(err_msg);
425 return false;
426 }
427
428 sqlite3_stmt* stmt = nullptr;
429 rc = sqlite3_prepare_v2(db,
430 "INSERT INTO migrations (name) VALUES (?)",
431 -1, &stmt, nullptr);
432 if (rc == SQLITE_OK) {
433 sqlite3_bind_text(stmt, 1, mig.name, -1, SQLITE_STATIC);
434 sqlite3_step(stmt);
435 sqlite3_finalize(stmt);
436 }
437 return true;
438}

462bool SqliteDatabase::run_migrations() {
463 char* err_msg = nullptr;
464 int rc = sqlite3_exec(db_,
465 "CREATE TABLE IF NOT EXISTS migrations ("
466 " id INTEGER PRIMARY KEY,"
467 " name TEXT UNIQUE,"
468 " applied_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP"
469 ")",
470 nullptr, nullptr, &err_msg);
471
472 if (rc != SQLITE_OK) {
473 logger->error("Failed to create migrations table: {}",
474 err_msg ? err_msg : "unknown");
475 sqlite3_free(err_msg);
476 return false;
477 }
478
479 auto applied = get_applied_migrations(db_);
480 for (const auto& mig : MIGRATIONS) {
481 if (is_applied(applied, mig.name)) continue;
482 if (!apply_migration(db_, mig)) return false;
483 }
484 return true;
485}
