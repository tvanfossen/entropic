72{

76 prompt += "\n\n" + base_prompt;

92 }
93 prompt += "\n\n" + format_tools(tool_jsons);
94 }

133bool ChatAdapter::is_response_complete(
134 const std::string& content,
135 const std::vector<ToolCall>& tool_calls) const
136{
137 // Has tool calls → not complete (needs execution)
138 if (!tool_calls.empty()) {
139 return false;
140 }
141
142 // Unclosed think block → still thinking
143 if (content.find("<think>") != std::string::npos &&
144 content.find("</think>") == std::string::npos)
145 {
146 return false;
147 }
148
149 // Strip think blocks and check for real content
150 std::string stripped = strip_think_blocks(content);
151 return !stripped.empty();
152}

183// ── Bare JSON parsing ──────────────────────────────────────
184
192std::vector<ToolCall> ChatAdapter::parse_bare_json_tool_calls(
193 const std::string& content) const
194{
195 std::vector<ToolCall> calls;
196 std::istringstream stream(content);
197 std::string line;
198
199 while (std::getline(stream, line)) {
200 // Trim
201 size_t start = line.find_first_not_of(" \t");
202 if (start == std::string::npos) continue;
203 std::string_view stripped(line.data() + start, line.size() - start);
204
205 if (stripped.front() != '{' || stripped.find("\"name\"") == std::string_view::npos) {
206 continue;
207 }
208
209 try {
210 auto j = nlohmann::json::parse(stripped);
211 if (j.contains("name")) {
212 ToolCall tc;
213 tc.id = generate_uuid();
214 tc.name = j["name"].get<std::string>();
215 auto args = j.value("arguments", j.value("parameters", nlohmann::json::object()));
216 for (auto& [k, v] : args.items()) {
217 tc.arguments[k] = v.dump();
218 }
219 calls.push_back(tc);
220 }
221 } catch (...) {
222 // Skip unparseable lines
223 }
224 }
225 return calls;
226}

258std::string ChatAdapter::strip_think_blocks(const std::string& content) const {
259 std::regex pattern(R"(<think>[\s\S]*?</think>)");
260 std::string result = std::regex_replace(content, pattern, "");
261
262 // Trim
263 size_t start = result.find_first_not_of(" \t\n\r");
264 if (start == std::string::npos) return "";
265 size_t end_pos = result.find_last_not_of(" \t\n\r");
266 return result.substr(start, end_pos - start + 1);
267}

281std::optional<ToolCall> ChatAdapter::try_recover_json(
282 const std::string& json_str) const
283{
284 // Fix trailing commas and single quotes
285 std::string fixed = std::regex_replace(json_str, std::regex(R"(,\s*\})"), "}");
286 fixed = std::regex_replace(fixed, std::regex(R"(,\s*\])"), "]");
287 std::replace(fixed.begin(), fixed.end(), '\'', '"');
288
289 logger->info("JSON recovery attempt: {} chars", json_str.size());
290 try {
291 auto j = nlohmann::json::parse(fixed);
292 if (j.contains("name")) {
293 ToolCall tc;
294 tc.id = generate_uuid();
295 tc.name = j["name"].get<std::string>();
296 auto args = j.value("arguments", j.value("parameters", nlohmann::json::object()));
297 for (auto& [k, v] : args.items()) {
298 tc.arguments[k] = v.dump();
299 }
300 return tc;
301 }
302 } catch (...) {
303 // Final fallback: extract name via regex
304 std::regex name_pattern(R"re("name"\s*:\s*"([^"]+)")re");
305 std::smatch match;
306 if (std::regex_search(json_str, match, name_pattern)) {
307 ToolCall tc;
308 tc.id = generate_uuid();
309 tc.name = match[1].str();
310 return tc;
311 }
312 }
313 return std::nullopt;
314}
