101static std::string parse_models_config(
102 ryml::ConstNodeRef node,
103 const BundledModels& registry,
104 ModelsConfig& config)
105{
106 extract(node, "default", config.default_tier);
107
108 if (node.has_child("router")) {
109 config.router.emplace();
110 auto err = parse_model_config(node["router"], registry,
111 *config.router);
112 if (!err.empty()) {
113 return "models.router: " + err;
114 }
115 }
116
117 for (auto child : node) {
118 std::string key = to_string(child.key());
119 if (key == "default" || key == "router") {
120 continue;
121 }
122 if (!child.is_map()) {
123 continue;
124 }
125
126 TierConfig tier;
127 if (config.tiers.count(key) > 0) {
128 tier = config.tiers[key];
129 }
130 auto err = parse_tier_config(child, registry, tier);
131 if (!err.empty()) {
132 return "models." + key + ": " + err;
133 }
134 config.tiers[key] = std::move(tier);
135 }
136
137 return "";
138}

391std::string parse_config_file(
392 const std::filesystem::path& path,
393 const BundledModels& registry,
394 ParsedConfig& config)
395{
396 std::string err;
397
398 auto content = read_file(path);
399 if (content.empty()) {
400 err = "cannot read config file: " + path.string();
401 }
402
403 ryml::Tree tree;
404 ryml::ConstNodeRef root;
405 if (err.empty()) {
406 tree = ryml::parse_in_arena(
407 ryml::to_csubstr(path.string()),
408 ryml::to_csubstr(content));
409 root = tree.rootref();
410 if (!root.is_map()) {
411 err = "config file root is not a YAML mapping: " + path.string();
412 }
413 }
414
415 if (err.empty() && root.has_child("models")) {
416 err = parse_models_config(root["models"], registry, config.models);
417 }
418 if (err.empty() && root.has_child("routing")) {
419 err = parse_routing_config(root["routing"], config.routing);
420 }
421 if (err.empty()) {
422 parse_optional_sections(root, config);
423 }
424
425 return err;
426}

438std::string load_config(
439 const std::filesystem::path& global_path,
440 const std::filesystem::path& project_path,
441 const BundledModels& registry,
442 ParsedConfig& config)
443{
444 bool have_config = false;
445
446 if (std::filesystem::exists(global_path)) {
447 s_log->info("Loading global config: {}", global_path.string());
448 auto err = parse_config_file(global_path, registry, config);
449 if (!err.empty()) {
450 return "global config: " + err;
451 }
452 have_config = true;
453 }
454
455 if (std::filesystem::exists(project_path)) {
456 s_log->info("Loading project config: {}", project_path.string());
457 auto err = parse_config_file(project_path, registry, config);
458 if (!err.empty()) {
459 return "project config: " + err;
460 }
461 have_config = true;
462 }
463
464 if (!have_config) {
465 auto data_dir = resolve_data_dir(config);
466 auto bundled = data_dir / "default_config.yaml";
467 if (std::filesystem::exists(bundled)) {
468 s_log->info("No user config — loading bundled default: {}",
469 bundled.string());
470 auto err = parse_config_file(bundled, registry, config);
471 if (!err.empty()) {
472 return "bundled default: " + err;
473 }
474 // Auto-create global config so this only happens once
475 if (!std::filesystem::exists(global_path)) {
476 auto parent = global_path.parent_path();
477 std::filesystem::create_directories(parent);
478 std::filesystem::copy_file(bundled, global_path);
479 s_log->info("Created {}", global_path.string());
480 }
481 } else {
482 s_log->warn("No config found (checked global, project, bundled)");
483 }
484 }
485
486 apply_env_overrides(config);
487
488 std::vector<std::string> warnings;
489 auto err = validate_config(config, warnings);
490 for (const auto& w : warnings) {
491 s_log->warn("{}", w);
492 }
493 return err;
494}

534static std::string parse_config_string(
535 const std::string& content,
536 const BundledModels& registry,
537 ParsedConfig& config)
538{
539 if (content.empty()) {
540 return "config string is empty";
541 }
542
543 ryml::Tree tree = ryml::parse_in_arena(
544 ryml::to_csubstr("<json>"),
545 ryml::to_csubstr(content));
546 auto root = tree.rootref();
547 if (!root.is_map()) {
548 return "config string root is not a mapping";
549 }
550
551 std::string err;
552 if (root.has_child("models")) {
553 err = parse_models_config(root["models"], registry, config.models);
554 }
555 if (err.empty() && root.has_child("routing")) {
556 err = parse_routing_config(root["routing"], config.routing);
557 }
558 if (err.empty()) {
559 parse_optional_sections(root, config);
560 }
561 return err;
562}
