49float lerp_channel(const uint8_t* data, int w,
50 int x0, int y0, int x1, int y1,
51 float fx, float fy, int ch) {
52 auto px = [data, w, ch](int x, int y) -> float {
53 return static_cast<float>(
54 data[(static_cast<size_t>(y) * static_cast<size_t>(w)
55 + static_cast<size_t>(x)) * 3
56 + static_cast<size_t>(ch)]);
57 };
58 float top = px(x0, y0) * (1.0f - fx) + px(x1, y0) * fx;
59 float bot = px(x0, y1) * (1.0f - fx) + px(x1, y1) * fx;
60 return top * (1.0f - fy) + bot * fy;
61}

74void interpolate_pixel(const uint8_t* src, int src_w, int src_h,
75 uint8_t* dst,
76 float src_x, float src_y) {
77 int x0 = std::max(0, static_cast<int>(src_x));
78 int x1 = std::min(src_w - 1, x0 + 1);
79 int y0 = std::max(0, static_cast<int>(src_y));
80 int y1 = std::min(src_h - 1, y0 + 1);
81 float fx = src_x - static_cast<float>(x0);
82 float fy = src_y - static_cast<float>(y0);
83
84 for (int ch = 0; ch < 3; ++ch) {
85 float val = lerp_channel(src, src_w, x0, y0, x1, y1, fx, fy, ch);
86 dst[ch] = static_cast<uint8_t>(std::clamp(val, 0.0f, 255.0f));
87 }
88}

98void bilinear_resize(entropic::PreprocessedImage& img,
99 int new_w, int new_h) {
100 std::vector<uint8_t> out(
101 static_cast<size_t>(new_w) * static_cast<size_t>(new_h) * 3);
102
103 float x_ratio = static_cast<float>(img.width) / static_cast<float>(new_w);
104 float y_ratio = static_cast<float>(img.height) / static_cast<float>(new_h);
105
106 for (int y = 0; y < new_h; ++y) {
107 float sy = (static_cast<float>(y) + 0.5f) * y_ratio - 0.5f;
108 for (int x = 0; x < new_w; ++x) {
109 float sx = (static_cast<float>(x) + 0.5f) * x_ratio - 0.5f;
110 size_t dst_off = (static_cast<size_t>(y)
111 * static_cast<size_t>(new_w)
112 + static_cast<size_t>(x)) * 3;
113 interpolate_pixel(img.pixel_data.data(),
114 img.width, img.height,
115 out.data() + dst_off, sx, sy);
116 }
117 }
118
119 img.pixel_data = std::move(out);
120 img.width = new_w;
121 img.height = new_h;
122}

144PreprocessedImage ImagePreprocessor::preprocess_file(
145 const std::filesystem::path& path) {
146 if (!std::filesystem::exists(path)) {
147 logger->error("Image file not found: {}", path.string());
148 throw std::runtime_error(
149 "Image file not found: " + path.string());
150 }
151
152 auto file_size = std::filesystem::file_size(path);
153 if (file_size > config_.max_file_size) {
154 logger->error("Image file too large: {} bytes (max {})",
155 file_size, config_.max_file_size);
156 throw std::runtime_error(
157 "Image exceeds max file size ("
158 + std::to_string(file_size) + " > "
159 + std::to_string(config_.max_file_size) + ")");
160 }
161
162 std::ifstream file(path, std::ios::binary);
163 if (!file) {
164 throw std::runtime_error(
165 "Cannot open image file: " + path.string());
166 }
167
168 std::vector<uint8_t> buf(static_cast<size_t>(file_size));
169 file.read(reinterpret_cast<char*>(buf.data()),
170 static_cast<std::streamsize>(file_size));
171
172 auto img = decode(buf.data(), buf.size(), path.string());
173 img.source_path = path.string();
174 resize_if_needed(img);
175
176 logger->info("Preprocessed image {}: {}x{} ({} bytes)",
177 path.string(), img.width, img.height,
178 img.pixel_data.size());
179 return img;
180}

214PreprocessedImage ImagePreprocessor::decode(
215 const uint8_t* data, size_t len,
216 const std::string& source) {
217 int w = 0;
218 int h = 0;
219 int ch = 0;
220 auto* pixels = stbi_load_from_memory(
221 data, static_cast<int>(len), &w, &h, &ch, 3);
222
223 if (pixels == nullptr) {
224 logger->error("Failed to decode image '{}': {}",
225 source, stbi_failure_reason());
226 throw std::runtime_error(
227 "Unsupported image format or decode error: " + source);
228 }
229
230 PreprocessedImage img;
231 img.width = w;
232 img.height = h;
233 img.channels = 3;
234 size_t data_size = static_cast<size_t>(w) * static_cast<size_t>(h) * 3;
235 img.pixel_data.assign(pixels, pixels + data_size);
236 stbi_image_free(pixels);
237
238 return img;
239}

252void ImagePreprocessor::resize_if_needed(PreprocessedImage& img) {
253 if (img.width <= config_.max_width
254 && img.height <= config_.max_height) {
255 return;
256 }
257
258 float scale_w = static_cast<float>(config_.max_width)
259 / static_cast<float>(img.width);
260 float scale_h = static_cast<float>(config_.max_height)
261 / static_cast<float>(img.height);
262
263 float scale = config_.preserve_aspect
264 ? std::min(scale_w, scale_h)
265 : 1.0f; // unused, but satisfies compiler
266
267 if (!config_.preserve_aspect) {
268 scale_w = std::min(scale_w, 1.0f);
269 scale_h = std::min(scale_h, 1.0f);
270 } else {
271 scale_w = scale;
272 scale_h = scale;
273 }
274
275 int new_w = std::max(1, static_cast<int>(
276 static_cast<float>(img.width) * scale_w));
277 int new_h = std::max(1, static_cast<int>(
278 static_cast<float>(img.height) * scale_h));
279
280 bilinear_resize(img, new_w, new_h);
281}
