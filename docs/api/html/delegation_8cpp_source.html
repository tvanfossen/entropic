91DelegationResult DelegationManager::execute_delegation(
92 LoopContext& parent_ctx,
93 const std::string& target_tier,
94 const std::string& task,
95 std::optional<int> max_turns) {
96
97 logger->info("Delegation: target_tier='{}', task='{}', depth={}",
98 target_tier, task,
99 parent_ctx.delegation_depth + 1);
100 auto info = tier_res_.resolve_tier
101 ? tier_res_.resolve_tier(target_tier, tier_res_.user_data)
102 : ChildContextInfo{};
103
104 if (!info.valid) {
105 logger->error("Tier '{}' not found", target_tier);
106 return {"Unknown tier: " + target_tier, false, target_tier, task};
107 }
108
109 auto child_ctx = build_child_context(parent_ctx, info, task);
110
111 // Create per-delegation worktree
112 std::optional<WorktreeInfo> wt_info;
113 if (worktree_mgr_) {
114 worktree_mgr_->ensure_develop();
115 wt_info = worktree_mgr_->create_worktree(
116 "d" + std::to_string(parent_ctx.delegation_depth + 1),
117 target_tier);
118 }
119
120 DelegationResult result;
121
122 // Scoped worktree swap (RAII restores on exit)
123 if (wt_info && swap_dir_fn_ != nullptr) {
124 ScopedWorktree scope(swap_dir_fn_, swap_dir_data_,
125 wt_info->path, repo_dir_);
126 result = run_child(child_ctx, target_tier, task, max_turns);
127 } else {
128 result = run_child(child_ctx, target_tier, task, max_turns);
129 }
130
131 finalize_worktree(wt_info, result);
132 return result;
133}

146static std::string pipeline_context(
147 size_t stage_idx, size_t total,
148 const std::vector<std::string>& stages) {
149 std::string ctx = "[PIPELINE CONTEXT]\n";
150 ctx += "Stage " + std::to_string(stage_idx + 1) +
151 " of " + std::to_string(total) + "\n";
152 ctx += "Role: " + stages[stage_idx] + "\n";
153 ctx += "Stay within your role. Do not perform work "

167DelegationResult DelegationManager::execute_pipeline(
168 LoopContext& parent_ctx,
169 const std::vector<std::string>& stages,
170 const std::string& task) {
171
172 logger->info("Pipeline: {} stages, task='{}'", stages.size(), task);
173
174 // Create shared worktree for entire pipeline
175 std::optional<WorktreeInfo> shared_wt;
176 if (worktree_mgr_) {
177 worktree_mgr_->ensure_develop();
178 shared_wt = worktree_mgr_->create_worktree("pipeline", "shared");
179 }
180
181 DelegationResult last_result;
182 last_result.task = task;
183
184 for (size_t i = 0; i < stages.size(); ++i) {
185 std::string stage_task =
186 pipeline_context(i, stages.size(), stages) + task;
187
188 auto info = tier_res_.resolve_tier
189 ? tier_res_.resolve_tier(stages[i], tier_res_.user_data)
190 : ChildContextInfo{};
191
192 if (!info.valid) {
193 logger->error("Pipeline stage {}: tier '{}' not found",
194 i, stages[i]);
195 last_result.success = false;
196 last_result.summary = "Unknown tier: " + stages[i];
197 break;
198 }
199
200 auto child_ctx = build_child_context(parent_ctx, info, stage_task);
201
202 if (shared_wt && swap_dir_fn_ != nullptr) {
203 ScopedWorktree scope(swap_dir_fn_, swap_dir_data_,
204 shared_wt->path, repo_dir_);
205 last_result = run_child(
206 child_ctx, stages[i], stage_task, std::nullopt);
207 } else {
208 last_result = run_child(
209 child_ctx, stages[i], stage_task, std::nullopt);
210 }
211
212 logger->info("Pipeline stage {} ({}): {}",
213 i, stages[i],
214 last_result.success ? "complete" : "failed");
215
216 if (!last_result.success) {
217 break;
218 }
219 }
220
221 // Pipeline always merges (even on partial failure)
222 if (shared_wt && worktree_mgr_) {
223 worktree_mgr_->merge_worktree(*shared_wt);
224 }
225
226 return last_result;
227}

240LoopContext DelegationManager::build_child_context(
241 const LoopContext& parent_ctx,
242 const ChildContextInfo& info,
243 const std::string& task) {
244
245 LoopContext child;
246 child.delegation_depth = parent_ctx.delegation_depth + 1;
247 child.locked_tier = info.system_prompt.empty()
248 ? parent_ctx.locked_tier : "";
249 child.all_tools = info.tools;
250 child.active_phase = "default";
251
252 // System prompt as first message
253 Message sys;
254 sys.role = "system";
255 sys.content = info.system_prompt;
256 child.messages.push_back(std::move(sys));
257
258 // Task as user message (with completion instructions)
259 std::string user_content = task;
260 if (!info.completion_instructions.empty()) {
261 user_content += "\n\n" + info.completion_instructions;
262 }
263 Message user;
264 user.role = "user";
265 user.content = std::move(user_content);
266 child.messages.push_back(std::move(user));
267
268 return child;
269}

278std::string DelegationManager::extract_summary(
279 const LoopContext& child_ctx) const {
280 // Prefer explicit completion summary
281 auto it = child_ctx.metadata.find("explicit_completion_summary");
282 if (it != child_ctx.metadata.end() && !it->second.empty()) {
283 return it->second;
284 }
285
286 // Fall back to last assistant message
287 for (auto rit = child_ctx.messages.rbegin();
288 rit != child_ctx.messages.rend(); ++rit) {
289 if (rit->role == "assistant" && !rit->content.empty()) {
290 return rit->content;
291 }
292 }
293
294 return "(No response from delegate)";
295}

365DelegationResult DelegationManager::run_child(
366 LoopContext& child_ctx,
367 const std::string& target_tier,
368 const std::string& task,
369 std::optional<int> max_turns) {
370
371 // Save parent todo list
372 std::string saved_todo;
373 if (todo_callbacks_.save != nullptr) {
374 saved_todo = todo_callbacks_.save(todo_callbacks_.user_data);
375 }
376 if (todo_callbacks_.install_fresh != nullptr) {
377 todo_callbacks_.install_fresh(todo_callbacks_.user_data);
378 }
379
380 auto delegation_id = create_storage_record(
381 child_ctx, target_tier, task, max_turns);
382
383 logger->info("Running child loop: tier={} depth={}",
384 target_tier, child_ctx.delegation_depth);
385
386 if (run_child_fn_ != nullptr) {
387 run_child_fn_(child_ctx, run_child_data_);
388 }
389
390 // Restore parent todo list
391 if (todo_callbacks_.restore != nullptr && !saved_todo.empty()) {
392 todo_callbacks_.restore(saved_todo, todo_callbacks_.user_data);
393 }
394
395 DelegationResult result;
396 result.target_tier = target_tier;
397 result.task = task;
398 result.success = (child_ctx.state == AgentState::COMPLETE);
399 result.turns_used = child_ctx.metrics.iterations;
400 result.summary = extract_summary(child_ctx);
401 result.child_messages = std::move(child_ctx.messages);
402
403 complete_storage_record(delegation_id, result);
404
405 logger->info("Child loop done: tier={} success={} turns={}",
406 target_tier, result.success, result.turns_used);
407 return result;
408}
